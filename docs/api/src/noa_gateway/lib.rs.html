1//! NOA Gateway - programmable, policy-enforced multi-protocol entrypoint
2//!
3//! This crate models the high-level orchestration of the application gateway used
4//! across NOA ARK OS. The design emphasises:
5//! - Programmable routing covering GraphQL federation, gRPC proxying, and WebSocket multiplexing.
6//! - Unified authentication & authorisation that leverages the core security subsystem.
7//! - Rate limiting tied to agent/service identities sourced from the hive mind registry.
8//! - Distributed tracing and telemetry export compatible with OpenTelemetry pipelines.
9//!
10//! The implementation intentionally focuses on deterministic, testable behaviour
11//! so it can run in CI without external infrastructure.

34/// High-level request entering the gateway.
35#[derive(Debug, Clone)]
36pub struct GatewayRequest {
37    pub request_id: String,
38    pub user_id: security::UserId,
39    pub agent_id: Option<String>,
40    pub credentials: AuthCredentials,
41    pub protocol: Protocol,
42    pub payload: serde_json::Value,
43    pub required_permission: Permission,
44}

46/// Simplified response emitted by the gateway after routing.
47#[derive(Debug, Clone)]
48pub struct GatewayResponse {
49    pub request_id: String,
50    pub route_plan: RoutePlan,
51    pub policy_enforced: bool,
52    pub timestamp: DateTime<Utc>,
53}

55/// Core orchestrator wiring all gateway subsystems together.
56pub struct Gateway {
57    authenticator: UnifiedAuthenticator,
58    policy: PolicyEnforcer,
59    router: ProgrammableRouter,
60    rate_limiter: RateLimiter,
61    telemetry: TelemetrySink,
62}
63
64impl Gateway {
65    /// Create a new gateway instance with sensible defaults.
66    pub fn new(
67        authenticator: UnifiedAuthenticator,
68        policy: PolicyEnforcer,
69        router: ProgrammableRouter,
70        rate_limiter: RateLimiter,
71        telemetry: TelemetrySink,
72    ) -> Result<Self> {
73        Ok(Self {
74            authenticator,
75            policy,
76            router,
77            rate_limiter,
78            telemetry,
79        })
80    }
81
82    /// Helper constructor that loads the shared agent registry and builds supporting components.
83    pub fn with_defaults(registry: Arc<AgentRegistry>, telemetry: TelemetrySink) -> Result<Self> {
84        let authenticator = UnifiedAuthenticator::default();
85        let policy = PolicyEnforcer::new();
86        let router = ProgrammableRouter::default();
87        let rate_limiter = RateLimiter::new(RateLimiterConfig::default(), registry);
88        Self::new(authenticator, policy, router, rate_limiter, telemetry)
89    }
90
91    /// Handle an incoming request by applying authN/Z, rate limiting, routing and telemetry.
92    #[instrument(skip(self))]
93    pub fn handle_request(&self, request: GatewayRequest) -> Result<GatewayResponse> {
94        // Step 1 - authenticate
95        self.authenticator
96            .verify(&request.credentials, &request.agent_id)
97            .context("authentication failed")?;
98
99        // Step 2 - authorise via core security policies
100        let enforcement_context =
101            EnforcementContext::from_payload(&request.protocol, &request.payload);
102        self.policy
103            .enforce(
104                request.user_id,
105                request.required_permission.clone(),
106                &enforcement_context,
107            )
108            .context("policy enforcement failure")?;
109
110        // Step 3 - enforce rate limits for the linked agent/service
111        self.rate_limiter
112            .check(&request.agent_id)
113            .context("rate limit exceeded")?;
114
115        // Step 4 - compute programmable route plan
116        let route_plan = self.router.route(&request.protocol, &request.payload)?;
117
118        // Step 5 - emit telemetry covering traces + metrics snapshot
119        self.telemetry.record(TelemetryEvent::new(
120            request.request_id.clone(),
121            request.protocol.clone(),
122            route_plan.clone(),
123            request.agent_id.clone(),
124        ))?;
125
126        Ok(GatewayResponse {
127            request_id: request.request_id,
128            route_plan,
129            policy_enforced: true,
130            timestamp: Utc::now(),
131        })
132    }
133}

135/// Build a production-like gateway composed of workspace primitives.
136pub fn bootstrap_gateway() -> Result<Gateway> {
137    // Ensure the security subsystem is initialised so policy checks work.
138    security::init().map_err(|err| anyhow!("failed to init security: {}", err))?;
139
140    // Load agent metadata to power identity aware rate limiting.
141    let registry =
142        Arc::new(AgentRegistry::with_default_data().context("failed to load agent registry")?);
143
144    let telemetry = TelemetrySink::default();
145    Gateway::with_defaults(registry, telemetry)
146}

216    #[test]
217    fn rate_limiting_blocks_after_token_bucket_exhaustion() {
218        let _ = security::init();
219        let registry =
220            Arc::new(AgentRegistry::with_default_data().expect("agent registry should load"));
221        let agent = registry
222            .all()
223            .into_iter()
224            .next()
225            .expect("at least one agent available");
226
227        let mut layer_limits = HashMap::new();
228        layer_limits.insert(agent.layer.clone(), 1);
229
230        let limiter = RateLimiter::new(
231            RateLimiterConfig {
232                refill_interval: Duration::from_secs(60),
233                layer_limits,
234            },
235            registry,
236        );
237
238        let agent_id = Some(agent.agent_id.clone());
239        assert!(limiter.check(&agent_id).is_ok());
240        let err = limiter
241            .check(&agent_id)
242            .expect_err("second call should exceed configured limit");
243        assert!(matches!(err, RateLimitError::LimitExceeded(_)));
244    }
