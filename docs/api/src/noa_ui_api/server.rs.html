251    async fn start_workflow(
252        State(state): State<UiApiState>,
253        Json(request): Json<WorkflowStartRequest>,
254    ) -> Result<Json<WorkflowStartResponse>, (StatusCode, Json<ErrorResponse>)> {
255        if request.workflow.name.trim().is_empty() {
256            return Err(bad_request("workflow name cannot be empty".into()));
257        }
258        if request.workflow.stages.is_empty() {
259            return Err(bad_request("workflow requires at least one stage".into()));
260        }
261
262        let workflow = request.workflow;
263        let workflow_id = workflow.name.clone();
264        let engine = state.workflow_engine();
265
266        engine
267            .load_workflow(workflow.clone())
268            .map_err(|err| bad_request(format!("failed to load workflow: {err}")))?;
269
270        let first_stage_id = workflow
271            .stages
272            .first()
273            .map(|stage| slugify_stage(&stage.name))
274            .unwrap_or_else(|| workflow_id.clone());
275
276        let resume_token = Some(default_resume_token(&workflow_id, &first_stage_id));
277        let stages = workflow
278            .stages
279            .iter()
280            .map(|stage| WorkflowStageSnapshot {
281                id: slugify_stage(&stage.name),
282                name: stage.name.clone(),
283                state: StageState::Pending,
284            })
285            .collect();
286
287        let spawned_engine = engine.clone();
288        let execution_id = workflow_id.clone();
289        task::spawn(async move {
290            let _ = spawned_engine.execute(&execution_id);
291        });
292
293        Ok(Json(WorkflowStartResponse {
294            workflow_id,
295            state: WorkflowState::Pending,
296            resume_token,
297            stages,
298            started_at: Utc::now().to_rfc3339(),
299        }))
300    }

302    async fn upload_drop(
303        State(state): State<UiApiState>,
304        mut multipart: Multipart,
305    ) -> Result<Json<DropUploadResponse>, (StatusCode, Json<ErrorResponse>)> {
306        let mut drop_type_value: Option<String> = None;
307        let mut file_name: Option<String> = None;
308        let mut file_bytes: Option<Bytes> = None;
309
310        while let Some(field) = multipart
311            .next_field()
312            .await
313            .map_err(|err| bad_request(format!("failed to read multipart field: {err}")))?
314        {
315            match field.name() {
316                Some("type") | Some("drop_type") => {
317                    let value = field
318                        .text()
319                        .await
320                        .map_err(|err| bad_request(format!("invalid drop type field: {err}")))?;
321                    drop_type_value = Some(value);
322                }
323                Some("file") | Some("upload") => {
324                    file_name = field.file_name().map(|name| name.to_string());
325                    let bytes = field.bytes().await.map_err(|err| {
326                        internal_error(format!("failed to read upload contents: {err}"))
327                    })?;
328                    file_bytes = Some(bytes);
329                }
330                _ => {}
331            }
332        }
333
334        let drop_type_value = drop_type_value
335            .ok_or_else(|| bad_request("missing required form field: type".to_string()))?;
336        let file_bytes = file_bytes
337            .ok_or_else(|| bad_request("missing required form field: file".to_string()))?;
338
339        let (source_type, directory_name) = parse_drop_type(&drop_type_value)
340            .ok_or_else(|| bad_request(format!("unsupported drop type: {}", drop_type_value)))?;
341
342        let drop_root = state.drop_root().clone();
343        let target_dir = drop_root.join(directory_name);
344        fs::create_dir_all(&target_dir)
345            .await
346            .map_err(|err| internal_error(format!("failed to prepare drop directory: {err}")))?;
347
348        let sanitized_name = sanitize_file_name(file_name.as_deref());
349        let file_path = target_dir.join(&sanitized_name);
350        let file_bytes = file_bytes.to_vec();
351        fs::write(&file_path, &file_bytes)
352            .await
353            .map_err(|err| internal_error(format!("failed to persist upload: {err}")))?;
354
355        let mut metadata = HashMap::new();
356        metadata.insert("original_filename".to_string(), sanitized_name.clone());
357        metadata.insert("drop_type".to_string(), drop_type_value.clone());
358
359        let timestamp = Utc::now().timestamp();
360        let manifest = DropManifest {
361            name: sanitized_name.clone(),
362            source: "ui/upload".to_string(),
363            source_type,
364            timestamp: if timestamp >= 0 { timestamp as u64 } else { 0 },
365            priority: Priority::Normal,
366            metadata,
367        };
368
369        let registry = state.drop_registry();
370        let drop_id = registry
371            .register(file_path.clone(), manifest, None)
372            .map_err(|err| internal_error(format!("failed to register drop: {err}")))?;
373
374        let status = registry
375            .get_state(&drop_id)
376            .map(|state| format_crc_state(&state))
377            .unwrap_or_else(|| format_crc_state(&CRCState::Incoming));
378
379        let processing = prepare_upload_receipt(
380            &file_path,
381            &sanitized_name,
382            &drop_type_value,
383            &drop_root,
384            &drop_id,
385            &file_bytes,
386        )
387        .await
388        .map_err(|err| internal_error(format!("failed to process upload: {err}")))?;
389
390        Ok(Json(DropUploadResponse {
391            drop_id,
392            status,
393            cas_keys: processing.cas_keys,
394            receipt_path: processing.receipt_path,
395            receipt_url: processing.receipt_url,
396        }))
397    }
398}

448fn parse_drop_type(value: &str) -> Option<(SourceType, &'static str)> {
449    let normalized = value.trim().to_lowercase();
450    match normalized.as_str() {
451        "repo" | "repos" | "external_repo" => Some((SourceType::ExternalRepo, "repos")),
452        "fork" | "forks" => Some((SourceType::Fork, "forks")),
453        "mirror" | "mirrors" => Some((SourceType::Mirror, "mirrors")),
454        "stale" | "stale_codebase" => Some((SourceType::StaleCodebase, "stale")),
455        "internal" => Some((SourceType::Internal, "internal")),
456        _ => None,
457    }
458}

460fn sanitize_file_name(file_name: Option<&str>) -> String {
461    file_name
462        .and_then(|name| Path::new(name).file_name().and_then(|value| value.to_str()))
463        .filter(|name| !name.is_empty())
464        .map(|name| name.to_string())
465        .unwrap_or_else(|| format!("upload-{}.bin", Uuid::new_v4()))
466}

468/// Slugifies a stage name for use in URLs or identifiers.
469///
470/// This implementation **must** stay in sync with the TypeScript `slugifyStage` function.
471/// It lowercases, replaces non-alphanumerics with hyphens, and collapses consecutive hyphens.
472fn slugify_stage(name: &str) -> String {
473    name.to_lowercase()
474        .chars()
475        .map(|ch| if ch.is_ascii_alphanumeric() { ch } else { '-' })
476        .collect::<String>()
477        .split('-')
478        .filter(|s| !s.is_empty())
479        .collect::<Vec<_>>()
480        .join("-")
481}

527async fn prepare_upload_receipt(
528    file_path: &Path,
529    original_name: &str,
530    drop_type: &str,
531    drop_root: &Path,
532    drop_id: &str,
533    original_bytes: &[u8],
534) -> Result<UploadProcessingResult> {
535    let cas = Cas::default().context("initializing CAS")?;
536
537    let mut cas_objects = Vec::new();
538
539    let original_hash = cas
540        .put_bytes(original_bytes)
541        .context("storing upload in CAS")?;
542    let original_meta = fs::metadata(file_path)
543        .await
544        .context("reading upload metadata")?;
545    cas_objects.push(UploadReceiptEntry {
546        hash: original_hash,
547        path: relative_path(file_path, drop_root),
548        size: original_meta.len(),
549        kind: "upload".to_string(),
550    });
551
552    if let Some(kind) = detect_archive(original_name) {
553        let extracted = extract_archive(file_path, drop_root, drop_id, kind).await?;
554        for artifact in extracted {
555            let bytes = fs::read(&artifact)
556                .await
557                .with_context(|| format!("reading extracted artifact {}", artifact.display()))?;
558            let hash = cas
559                .put_bytes(&bytes)
560                .with_context(|| format!("storing extracted artifact {}", artifact.display()))?;
561            let meta = fs::metadata(&artifact)
562                .await
563                .with_context(|| format!("reading metadata for {}", artifact.display()))?;
564            cas_objects.push(UploadReceiptEntry {
565                hash,
566                path: relative_path(&artifact, drop_root),
567                size: meta.len(),
568                kind: "extracted".to_string(),
569            });
570        }
571    }
572
573    let mut graph = CRCGraph::new();
574    graph.add_node(GraphNode::new(
575        format!("digest::{}", drop_id),
576        NodeKind::Analyze,
577        Lane::Fast,
578    ));
579    let engine = Engine::new(graph);
580    let digest_dir = drop_root.join("receipts").join(drop_id).join("digest");
581    fs::create_dir_all(&digest_dir)
582        .await
583        .with_context(|| format!("creating digest directory {}", digest_dir.display()))?;
584    let summary = engine
585        .run(&digest_dir)
586        .await
587        .context("running CRC digest engine")?;
588
589    let receipt_dir = drop_root.join("receipts");
590    fs::create_dir_all(&receipt_dir)
591        .await
592        .with_context(|| format!("creating receipt directory {}", receipt_dir.display()))?;
593    let receipt_path = receipt_dir.join(format!("{drop_id}.receipt.json"));
594
595    let receipt = UploadReceiptDocument {
596        drop_id: drop_id.to_string(),
597        drop_type: drop_type.to_string(),
598        original_name: original_name.to_string(),
599        cas_objects: cas_objects.clone(),
600        digest: UploadReceiptDigest {
601            checkpoint: summary.checkpoint.to_string_lossy().to_string(),
602            executed_nodes: summary.executed.len(),
603        },
604        created_at: Utc::now().to_rfc3339(),
605    };
606
607    let serialized = serde_json::to_vec_pretty(&receipt).context("serializing upload receipt")?;
608    fs::write(&receipt_path, serialized)
609        .await
610        .with_context(|| format!("writing receipt {}", receipt_path.display()))?;
611
612    let cas_keys = cas_objects.iter().map(|entry| entry.hash.clone()).collect();
613    let receipt_path_str = receipt_path.to_string_lossy().to_string();
614    Ok(UploadProcessingResult {
615        cas_keys,
616        receipt_path: receipt_path_str.clone(),
617        receipt_url: Some(format!("file://{}", receipt_path_str)),
618    })
619}

621fn detect_archive(file_name: &str) -> Option<ArchiveKind> {
622    let lowered = file_name.to_lowercase();
623    if lowered.ends_with(".tar.gz") || lowered.ends_with(".tgz") {
624        Some(ArchiveKind::TarGz)
625    } else if lowered.ends_with(".tar") {
626        Some(ArchiveKind::Tar)
627    } else if lowered.ends_with(".zip") {
628        Some(ArchiveKind::Zip)
629    } else {
630        None
631    }
632}

634async fn extract_archive(
635    archive_path: &Path,
636    drop_root: &Path,
637    drop_id: &str,
638    kind: ArchiveKind,
639) -> Result<Vec<PathBuf>> {
640    let target = drop_root.join("extracted").join(drop_id);
641    fs::create_dir_all(&target)
642        .await
643        .with_context(|| format!("creating extraction directory {}", target.display()))?;
644    let archive_path = archive_path.to_path_buf();
645    let target_clone = target.clone();
646    let extracted = task::spawn_blocking(move || match kind {
647        ArchiveKind::Zip => extract_zip(&archive_path, &target_clone),
648        ArchiveKind::Tar => extract_tar(&archive_path, &target_clone, false),
649        ArchiveKind::TarGz => extract_tar(&archive_path, &target_clone, true),
650    })
651    .await
652    .context("joining archive extraction task")?;
653
654    let extracted = extracted?;
655    Ok(extracted)
656}

658fn extract_zip(archive_path: &Path, target: &Path) -> Result<Vec<PathBuf>> {
659    let file = std::fs::File::open(archive_path)
660        .with_context(|| format!("opening zip archive {}", archive_path.display()))?;
661    let mut archive = ZipArchive::new(file).context("parsing zip archive")?;
662    let mut extracted = Vec::new();
663    for index in 0..archive.len() {
664        let mut entry = archive.by_index(index).context("reading zip entry")?;
665        if entry.is_dir() {
666            continue;
667        }
668        let Some(path) = entry.enclosed_name().map(|path| path.to_owned()) else {
669            continue;
670        };
671        let out_path = target.join(path);
672        if let Some(parent) = out_path.parent() {
673            std::fs::create_dir_all(parent)
674                .with_context(|| format!("creating directory {}", parent.display()))?;
675        }
676        let mut output = std::fs::File::create(&out_path)
677            .with_context(|| format!("creating {}", out_path.display()))?;
678        std::io::copy(&mut entry, &mut output)
679            .with_context(|| format!("extracting {}", out_path.display()))?;
680        extracted.push(out_path);
681    }
682    Ok(extracted)
683}

685fn extract_tar(archive_path: &Path, target: &Path, gz: bool) -> Result<Vec<PathBuf>> {
686    let file = std::fs::File::open(archive_path)
687        .with_context(|| format!("opening tar archive {}", archive_path.display()))?;
688    let reader: Box<dyn std::io::Read> = if gz {
689        Box::new(GzDecoder::new(file))
690    } else {
691        Box::new(file)
692    };
693    let mut archive = tar::Archive::new(reader);
694    let mut extracted = Vec::new();
695    for entry in archive.entries().context("iterating tar entries")? {
696        let mut entry = entry.context("reading tar entry")?;
697        if entry.header().entry_type().is_dir() {
698            continue;
699        }
700        let relative = entry.path().context("reading tar entry path")?.into_owned();
701        if relative
702            .components()
703            .any(|component| matches!(component, std::path::Component::ParentDir))
704        {
705            continue;
706        }
707        let out_path = target.join(&relative);
708        if let Some(parent) = out_path.parent() {
709            std::fs::create_dir_all(parent)
710                .with_context(|| format!("creating directory {}", parent.display()))?;
711        }
712        entry
713            .unpack(&out_path)
714            .with_context(|| format!("extracting {}", out_path.display()))?;
715        extracted.push(out_path);
716    }
717    Ok(extracted)
718}

720fn relative_path(path: &Path, drop_root: &Path) -> String {
721    path.strip_prefix(drop_root)
722        .map(|p| p.to_string_lossy().to_string())
723        .unwrap_or_else(|_| path.to_string_lossy().to_string())
724}

726async fn handle_websocket(mut socket: WebSocket, bridge: SessionBridge) {
727    let mut events = bridge.subscribe();
728    while let Some(event) = events.next().await {
729        let Ok(event) = event.map(SessionBridge::map_event) else {
730            continue;
731        };
732
733        match serde_json::to_string(&event) {
734            Ok(payload) => {
735                if socket.send(Message::Text(payload)).await.is_err() {
736                    break;
737                }
738            }
739            Err(error) => {
740                let _ = socket
741                    .send(Message::Text(format!("{{\"error\":\"{}\"}}", error)))
742                    .await;
743            }
744        }
745    }
746}

861        Request::builder()
862            .method("POST")
863            .uri("/ui/drop-in/upload")
864            .header(
865                "content-type",

867            )

869            .unwrap()
870    }

872    fn build_multipart_body(boundary: &str, parts: &[(&str, &str, Option<&str>)]) -> String {
873        let mut body = String::new();
874        for (name, value, filename) in parts {
875            body.push_str(&format!("--{boundary}\r\n"));
876            if let Some(filename) = filename {
877                body.push_str(&format!(
878                    "Content-Disposition: form-data; name=\"{name}\"; filename=\"{filename}\"\r\n"
879                ));
880                body.push_str("Content-Type: application/octet-stream\r\n\r\n");
881                body.push_str(value);
882                body.push_str("\r\n");
883            } else {
884                body.push_str(&format!(
885                    "Content-Disposition: form-data; name=\"{name}\"\r\n\r\n{value}\r\n"
886                ));
887            }
888        }
889        body.push_str(&format!("--{boundary}--\r\n"));
890        body
891    }
