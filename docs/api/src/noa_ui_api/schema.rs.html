29#[derive(Debug, Clone, Serialize_repr, Deserialize)]
30#[repr(u8)]
31pub enum LayoutSlot {
32    Header = 1,
33    Main = 2,
34    Footer = 3,
35}
36
37impl fmt::Display for LayoutSlot {
38    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
39        let label = match self {
40            LayoutSlot::Header => "header",
41            LayoutSlot::Main => "main",
42            LayoutSlot::Footer => "footer",
43        };
44
45        f.write_str(label)
46    }
47}
