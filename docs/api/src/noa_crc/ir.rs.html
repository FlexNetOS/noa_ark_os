91#[derive(Debug, Clone, Serialize, Deserialize)]
92pub struct NodeState {
93    pub id: NodeId,
94    pub kind: crate::graph::NodeKind,
95    pub lane: Lane,
96    pub facets: Vec<Facet>,
97    pub io: NodeIo,
98    pub dependencies: BTreeSet<NodeId>,
99    pub cache_key: String,
100}
101
102impl NodeState {
103    pub fn compute_cache_key(&self) -> String {
104        let mut hasher = blake3::Hasher::new();
105        hasher.update(self.id.0.as_bytes());
106        hasher.update(&[self.kind as u8]);
107        hasher.update(&[self.lane as u8]);
108        for dep in &self.dependencies {
109            hasher.update(dep.0.as_bytes());
110        }
111        for (name, handle) in &self.io.inputs {
112            hasher.update(name.as_bytes());
113            hasher.update(handle.artifact.hash.as_bytes());
114        }
115        hasher.finalize().to_hex().to_string()
116    }
117}

140    #[test]
141    fn cache_key_changes_with_inputs() {
142        let mut node = NodeState {
143            id: NodeId::new(),
144            kind: crate::graph::NodeKind::Analyze,
145            lane: Lane::Fast,
146            facets: vec![],
147            io: NodeIo::new(),
148            dependencies: BTreeSet::new(),
149            cache_key: String::new(),
150        };
151        let provenance = Provenance {
152            origin: "unit-test".into(),
153            description: None,
154            captured_at: chrono::Utc::now(),
155            trust_score: 1.0,
156        };
157        node.io.inputs.insert(
158            "input".into(),
159            DataHandle {
160                key: "input".into(),
161                artifact: ArtifactRef::new(None, b"hello", None),
162                facets: vec![],
163                provenance: provenance.clone(),
164            },
165        );
166        let key1 = node.compute_cache_key();
167        node.io.inputs.get_mut("input").unwrap().artifact = ArtifactRef::new(None, b"world", None);
168        let key2 = node.compute_cache_key();
169        assert_ne!(key1, key2);
170    }
