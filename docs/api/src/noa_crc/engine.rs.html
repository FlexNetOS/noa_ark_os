12#[derive(Default)]
13struct CacheEntry {
14    outputs: BTreeMap<String, crate::ir::DataHandle>,
15}

17#[derive(Default)]
18struct NodeCache {
19    entries: std::collections::HashMap<NodeId, CacheEntry>,
20}
21
22impl NodeCache {
23    fn get(&self, id: &NodeId) -> Option<&CacheEntry> {
24        self.entries.get(id)
25    }
26
27    fn insert(&mut self, id: NodeId, entry: CacheEntry) {
28        self.entries.insert(id, entry);
29    }
30}

46    pub async fn run(&self, checkpoint_dir: &Path) -> Result<ExecutionSummary> {
47        fs::create_dir_all(checkpoint_dir)?;
48        let mut executed = Vec::new();
49        for id in self.graph.topo_order()? {
50            let node = self
51                .graph
52                .node(&id)
53                .cloned()
54                .ok_or_else(|| anyhow!("node missing"))?;
55            let state = self.execute_node(node).await?;
56            executed.push(state.clone());
57        }
58        let snapshot = Snapshot::new(executed.clone(), Some("run completion".into()));
59        self.persist_snapshot(checkpoint_dir, &snapshot)?;
60        Ok(ExecutionSummary {
61            executed,
62            checkpoint: checkpoint_dir.to_path_buf(),
63        })
64    }
65
66    async fn execute_node(&self, node: GraphNode) -> Result<NodeState> {
67        let cache = self.cache.lock().await;
68        if let Some(entry) = cache.get(&node.id) {
69            let state = NodeState {
70                id: node.id.clone(),
71                kind: node.kind,
72                lane: node.lane,
73                facets: vec![],
74                io: crate::ir::NodeIo {
75                    inputs: Default::default(),
76                    outputs: entry.outputs.clone(),
77                },
78                dependencies: Default::default(),
79                cache_key: blake3::hash(node.name.as_bytes()).to_hex().to_string(),
80            };
81            return Ok(state);
82        }
83        drop(cache);
84
85        // Simulate execution based on kind.
86        let mut outputs = BTreeMap::new();
87        let provenance = crate::ir::Provenance {
88            origin: format!("{}", node.kind as u8),
89            description: Some(format!("Executed {}", node.name)),
90            captured_at: chrono::Utc::now(),
91            trust_score: match node.kind {
92                NodeKind::Analyze | NodeKind::Verify => 0.9,
93                NodeKind::Decide => 0.8,
94                NodeKind::Transform => 0.7,
95                NodeKind::Persist => 1.0,
96            },
97        };
98        outputs.insert(
99            "stdout".into(),
100            crate::ir::DataHandle {
101                key: "stdout".into(),
102                artifact: crate::ir::ArtifactRef::new(
103                    None,
104                    node.name.as_bytes(),
105                    Some("text/plain".into()),
106                ),
107                facets: vec![crate::ir::Facet {
108                    kind: match node.kind {
109                        NodeKind::Analyze => crate::ir::FacetKind::Analysis,
110                        NodeKind::Decide => crate::ir::FacetKind::Decision,
111                        NodeKind::Transform => crate::ir::FacetKind::Transformation,
112                        NodeKind::Verify => crate::ir::FacetKind::Verification,
113                        NodeKind::Persist => crate::ir::FacetKind::Persistence,
114                    },
115                    lane: node.lane,
116                    metadata: Default::default(),
117                }],
118                provenance,
119            },
120        );
121
122        let mut cache = self.cache.lock().await;
123        cache.insert(
124            node.id.clone(),
125            CacheEntry {
126                outputs: outputs.clone(),
127            },
128        );
129        drop(cache);
130
131        Ok(NodeState {
132            id: node.id,
133            kind: node.kind,
134            lane: node.lane,
135            facets: vec![],
136            io: crate::ir::NodeIo {
137                inputs: Default::default(),
138                outputs,
139            },
140            dependencies: Default::default(),
141            cache_key: blake3::hash(node.name.as_bytes()).to_hex().to_string(),
142        })
143    }
144
145    fn persist_snapshot(&self, checkpoint_dir: &Path, snapshot: &Snapshot) -> Result<()> {
146        let path = checkpoint_dir.join(format!("snapshot-{}.json", chrono::Utc::now().timestamp()));
147        let data = serde_json::to_vec_pretty(snapshot)?;
148        fs::write(path, data)?;
149        Ok(())
150    }
151}

165    async fn cache_hits_skip_execution() {
166        let mut graph = CRCGraph::new();

168        let engine = Engine::new(graph);
169        let tmp = tempfile::tempdir().unwrap();
170        let first = engine.run(tmp.path()).await.unwrap();
171        assert_eq!(first.executed.len(), 1);
172        let second = engine.run(tmp.path()).await.unwrap();
