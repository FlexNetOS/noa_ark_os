9/// Target hardware profile the CRC automation optimizes for.
10#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
11#[serde(rename_all = "lowercase")]
12pub enum TargetProfile {
13    Edge,
14    Server,
15}
16
17impl TargetProfile {
18    pub fn artifact_directory(&self) -> &'static str {
19        match self {
20            TargetProfile::Edge => "edge",
21            TargetProfile::Server => "server",
22        }
23    }
24
25    pub fn optimization_flags(&self) -> Vec<&'static str> {
26        match self {
27            TargetProfile::Edge => vec![
28                "-C",
29                "opt-level=z",
30                "-C",
31                "lto=fat",
32                "-C",
33                "codegen-units=1",
34            ],
35            TargetProfile::Server => vec!["-C", "opt-level=3", "-C", "lto=thin"],
36        }
37    }
38
39    pub fn feature_gates(&self) -> Vec<&'static str> {
40        match self {
41            TargetProfile::Edge => vec!["low-power", "quantized-kernels"],
42            TargetProfile::Server => vec!["throughput", "gpu-offload"],
43        }
44    }
45
46    pub fn description(&self) -> &'static str {
47        match self {
48            TargetProfile::Edge => "Edge devices prioritizing thermal and power limits",
49            TargetProfile::Server => "Throughput-optimized server with abundant resources",
50        }
51    }
52}

54/// Manifest describing an optimized build that CRC produced.
55#[derive(Debug, Clone, Serialize, Deserialize)]
56pub struct BuildManifest {
57    pub drop_id: String,
58    pub profile: TargetProfile,
59    pub source_path: String,
60    pub optimization_flags: Vec<String>,
61    pub feature_gates: Vec<String>,
62    pub timestamp: u64,
63}
64
65/// Materialized artifact location for a compiled profile.
66#[derive(Debug, Clone, Serialize, Deserialize)]
67pub struct BuildArtifact {
68    pub manifest: BuildManifest,
69    pub artifact_path: PathBuf,
70    pub notes: Vec<String>,
71}
72
73/// Generate optimized builds for each target profile and persist manifests.
74pub async fn generate_optimized_builds(
75    drop_id: &str,
76    source_path: &Path,
77    artifact_root: &Path,
78) -> Result<Vec<BuildArtifact>> {
79    let mut artifacts = Vec::new();
80
81    for profile in [TargetProfile::Edge, TargetProfile::Server] {
82        let profile_root = artifact_root
83            .join(profile.artifact_directory())
84            .join(drop_id);
85        fs::create_dir_all(&profile_root).await?;
86
87        let manifest = BuildManifest {
88            drop_id: drop_id.to_string(),
89            profile,
90            source_path: source_path.display().to_string(),
91            optimization_flags: profile
92                .optimization_flags()
93                .into_iter()
94                .map(|flag| flag.to_string())
95                .collect(),
96            feature_gates: profile
97                .feature_gates()
98                .into_iter()
99                .map(|flag| flag.to_string())
100                .collect(),
101            timestamp: SystemTime::now()
102                .duration_since(UNIX_EPOCH)
103                .unwrap_or_default()
104                .as_secs(),
105        };
106
107        let manifest_yaml = serde_yaml::to_string(&manifest)?;
108        let manifest_path = profile_root.join("manifest.yaml");
109        fs::write(&manifest_path, manifest_yaml).await?;
110
111        let flags_path = profile_root.join("flags.txt");
112        let flags_body = manifest.optimization_flags.join(" ");
113        fs::write(&flags_path, flags_body).await?;
114
115        let notes = vec![
116            format!("Optimizations tuned for {}", profile.description()),
117            "Artifacts are ready for downstream packaging".to_string(),
118        ];
119
120        artifacts.push(BuildArtifact {
121            manifest,
122            artifact_path: profile_root,
123            notes,
124        });
125    }
126
127    Ok(artifacts)
128}
