15fn main() -> Result<()> {
16    let (root, once) = parse_args(env::args().skip(1));
17    if once {
18        run_once(&root)
19    } else {
20        run_watch(root)
21    }
22}
23
24fn parse_args<I>(args: I) -> (PathBuf, bool)
25where
26    I: IntoIterator<Item = String>,
27{
28    let mut root = PathBuf::from(".");
29    let mut once = false;
30    for arg in args {
31        if arg == "--once" {
32            once = true;
33        } else if arg.starts_with("--") {
34            eprintln!("[symbol-graph] ignoring unknown flag {arg}");
35        } else {
36            root = PathBuf::from(arg);
37        }
38    }
39    (root, once)
40}

42fn run_once(root: &Path) -> Result<()> {
43    let mut previous = load_existing_graph(root)?;
44    let new_graph = rebuild_graph(root)?;
45    let diff = NotebookMetadataDiff::from_graphs(&previous, &new_graph);
46    if diff.has_changes() {
47        write_diff(root, &diff)?;
48    }
49    previous = new_graph;
50    persist_state(root, &previous)?;
51    Ok(())
52}
53
54fn run_watch(root: PathBuf) -> Result<()> {
55    let mut previous = load_existing_graph(&root)?;
56    let (tx, rx) = channel();
57    let mut watcher = RecommendedWatcher::new(tx, Config::default())
58        .with_context(|| "failed to start filesystem watcher")?;
59    watcher
60        .watch(&root, RecursiveMode::Recursive)
61        .with_context(|| format!("failed to watch {}", root.display()))?;
62
63    for event in rx {
64        match event {
65            Ok(event) => {
66                if should_process(&event) {
67                    if let Err(err) = handle_event(&root, &mut previous) {
68                        eprintln!("[symbol-graph] watcher error: {err}");
69                    }
70                }
71            }
72            Err(err) => eprintln!("[symbol-graph] watcher channel error: {err}"),
73        }
74    }
75    Ok(())
76}

78fn handle_event(root: &Path, previous: &mut SymbolGraph) -> Result<()> {
79    let new_graph = rebuild_graph(root)?;
80    let diff = NotebookMetadataDiff::from_graphs(previous, &new_graph);
81    if diff.has_changes() {
82        write_diff(root, &diff)?;
83    }
84    *previous = new_graph;
85    persist_state(root, previous)?;
86    Ok(())
87}

89fn should_process(event: &Event) -> bool {
90    if matches!(event.kind, EventKind::Other) {
91        return false;
92    }
93    if matches!(event.kind, EventKind::Modify(ModifyKind::Metadata(_))) {
94        return false;
95    }
96    event
97        .paths
98        .iter()
99        .any(|path| is_relevant_file(path) && !is_workspace_path(path))
100}
101
102fn is_relevant_file(path: &Path) -> bool {
103    matches!(
104        path.extension().and_then(|ext| ext.to_str()),
105        Some("rs" | "ts" | "tsx")
106    )
107}
108
109fn is_workspace_path(path: &Path) -> bool {
110    path.components().any(|component| {
111        matches!(
112            component.as_os_str().to_str(),
113            Some(".workspace" | "target" | "node_modules")
114        )
115    })
116}

118fn rebuild_graph(root: &Path) -> Result<SymbolGraph> {
119    SymbolGraphBuilder::new(root)
120        .index()
121        .with_context(|| format!("failed to rebuild symbol graph for {}", root.display()))
122}

124fn load_existing_graph(root: &Path) -> Result<SymbolGraph> {
125    let store_root = symbol_store_root(root);
126    Ok(SymbolGraph::load(&store_root).unwrap_or_default())
127}

129fn persist_state(root: &Path, graph: &SymbolGraph) -> Result<()> {
130    let store_root = symbol_store_root(root);
131    std::fs::create_dir_all(&store_root).with_context(|| {
132        format!(
133            "failed to create symbol graph store at {}",
134            store_root.display()
135        )
136    })?;
137    let nodes_path = store_root.join("notebook_state.json");
138    std::fs::write(&nodes_path, to_string_pretty(graph)?).with_context(|| {
139        format!(
140            "failed to record notebook state at {}",
141            nodes_path.display()
142        )
143    })
144}

146fn write_diff(root: &Path, diff: &NotebookMetadataDiff) -> Result<PathBuf> {
147    let diff_root = root.join(".workspace").join("notebook_sync").join("diffs");
148    std::fs::create_dir_all(&diff_root)
149        .with_context(|| format!("failed to create diff directory at {}", diff_root.display()))?;
150    let timestamp = SystemTime::now()
151        .duration_since(UNIX_EPOCH)
152        .unwrap_or_default()
153        .as_secs();
154    let mut candidate = diff_root.join(format!("diff-{timestamp}.json"));
155    let mut counter = 0u32;
156    while candidate.exists() {
157        counter += 1;
158        candidate = diff_root.join(format!("diff-{timestamp}-{counter}.json"));
159    }
160    std::fs::write(&candidate, to_string_pretty(diff)?).with_context(|| {
161        format!(
162            "failed to write notebook metadata diff to {}",
163            candidate.display()
164        )
165    })?;
166    Ok(candidate)
167}

169fn symbol_store_root(root: &Path) -> PathBuf {
170    root.join(".workspace").join("indexes").join("symbol_graph")
171}

176    use std::fs;

179    #[test]
180    fn parse_args_supports_root_and_once() {
181        let (root, once) = parse_args(vec!["./workspace".into(), "--once".into()]);
182        assert_eq!(root, PathBuf::from("./workspace"));
183        assert!(once);
184    }
185
186    #[test]
187    fn write_diff_creates_unique_files() {
188        let temp = tempdir().unwrap();
189        let diff = NotebookMetadataDiff::empty();
190        let first = write_diff(temp.path(), &diff).unwrap();
191        let second = write_diff(temp.path(), &diff).unwrap();
192        assert!(first.exists());
193        assert!(second.exists());
194        assert_ne!(first, second);
195    }
196
197    #[test]
198    fn handle_event_writes_diff_when_changes_detected() {
199        let temp = tempdir().unwrap();
200        let root = temp.path();
201        fs::create_dir_all(root.join("src")).unwrap();
202        fs::write(root.join("src/lib.rs"), "pub fn example() {}").unwrap();
203
204        let mut previous = SymbolGraph::default();
205        handle_event(root, &mut previous).unwrap();
206        let diff_root = root.join(".workspace/notebook_sync/diffs");
207        assert!(diff_root.exists());
208        let entries: Vec<_> = fs::read_dir(diff_root).unwrap().collect();
209        assert!(!entries.is_empty());
