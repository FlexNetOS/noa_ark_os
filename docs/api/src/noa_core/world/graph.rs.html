9#[derive(Debug, Error)]
10pub enum WorldGraphError {
11    #[error("io error while reading world graph: {0}")]
12    Io(#[from] std::io::Error),
13    #[error("failed to parse world graph: {0}")]
14    Parse(#[from] serde_json::Error),
15    #[error("world graph validation error: {0}")]
16    Validation(String),
17}

42impl Node {

70#[derive(Debug, Clone, Deserialize, Serialize)]

78impl WorldGraph {
79    pub fn load_from_reader<R: Read>(reader: R) -> Result<Self, WorldGraphError> {
80        let graph: WorldGraph = serde_json::from_reader(reader)?;
81        graph.validate()?;
82        Ok(graph)
83    }
84
85    pub fn load_from_path<P: AsRef<Path>>(path: P) -> Result<Self, WorldGraphError> {
86        let file = File::open(path)?;
87        let reader = BufReader::new(file);
88        Self::load_from_reader(reader)
89    }
90
91    pub fn load_default() -> Result<Self, WorldGraphError> {
92        let repo_root = Self::repo_root();
93        let graph_path = repo_root.join("kernel/world/world.graph.json");
94        Self::load_from_path(graph_path)
95    }
96
97    pub fn repo_root() -> PathBuf {
98        let manifest_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"));
99        manifest_dir
100            .parent()
101            .map(Path::to_path_buf)
102            .unwrap_or(manifest_dir)
103    }
104
105    pub fn validate(&self) -> Result<(), WorldGraphError> {
106        let mut ids = HashSet::new();
107        for node in &self.nodes {
108            if !ids.insert(&node.id) {
109                return Err(WorldGraphError::Validation(format!(
110                    "duplicate node id detected: {}",
111                    node.id
112                )));
113            }
114            if node.path.trim().is_empty() {
115                return Err(WorldGraphError::Validation(format!(
116                    "node {} has an empty path",
117                    node.id
118                )));
119            }
120        }
121
122        let node_ids: HashSet<_> = self.nodes.iter().map(|n| n.id.as_str()).collect();
123        for edge in &self.edges {
124            if !node_ids.contains(edge.source.as_str()) {
125                return Err(WorldGraphError::Validation(format!(
126                    "edge references missing source node {}",
127                    edge.source
128                )));
129            }
130            if !node_ids.contains(edge.target.as_str()) {
131                return Err(WorldGraphError::Validation(format!(
132                    "edge references missing target node {}",
133                    edge.target
134                )));
135            }
136        }
137
138        Ok(())
139    }
140}
