301            return false;
302        }
303        if self.min_trust_score < constraints.min_trust_score {

311        }
312        // QoS tier must be at least as high as requested
313        if self.qos_tier.priority() < constraints.qos_tier.priority() {
314            return false;
315        }
316        true
317    }

382/// Compiles business intents into circuit layouts and records the resulting audit log.
383#[derive(Debug, Default)]
384pub struct PolicyCompiler {
385    verification_log: RwLock<Vec<String>>,
386}
387
388impl PolicyCompiler {
389    pub fn new() -> Self {
390        Self::default()
391    }
392
393    pub fn compile(
394        &self,
395        intent: &Intent,
396        candidate_connectors: &[String],
397        twin: &SemanticTwin,
398    ) -> Result<CircuitLayout, GatewayError> {
399        if candidate_connectors.is_empty() {
400            return Err(GatewayError::NoRouteFound);
401        }
402
403        if !twin.evaluate_intent(intent, candidate_connectors) {
404            return Err(GatewayError::VerificationFailed(
405                "knowledge graph rejected connector combination".into(),
406            ));
407        }
408
409        let optimized_latency_ms = candidate_connectors.len().min(3) as u32
410            * intent.constraints.max_latency_ms
411            / (candidate_connectors.len().max(1) as u32);
412
413        let layout = CircuitLayout {
414            connectors: candidate_connectors.to_vec(),
415            optimized_latency_ms,
416            secure: false,
417            proofs: vec![format!("intent:{}", intent.description)],
418        };
419
420        {
421            let mut log = self
422                .verification_log
423                .write()
424                .map_err(|_| GatewayError::Poisoned("policy_compiler"))?;
425            log.push(format!(
426                "compiled:{} targets:{}",
427                intent.description,
428                candidate_connectors.join(",")
429            ));
430        }
431
432        Ok(layout)
433    }
434
435    pub fn verification_log(&self) -> Result<Vec<String>, GatewayError> {
436        self.verification_log
437            .read()
438            .map(|log| log.clone())
439            .map_err(|_| GatewayError::Poisoned("policy_compiler"))
440    }
441}

493        self.mutation_history

505        fingerprint.sort();
506        let mutation = format!(
507            "adaptive-{}-{}",
508            fingerprint.join("+"),
509            self.mutation_history.len() + 1
510        );
511        self.performance_score = (self.performance_score + 0.12).min(1.0);
512        self.mutation_history.push(mutation.clone());
513        if self.mutation_history.len() > 16 {
514            self.mutation_history.remove(0);
515        }
516        Some(mutation)

520#[derive(Debug, Clone)]
521pub struct ExecutionSchematic {
522    pub nodes: Vec<String>,
523    pub edges: Vec<(String, String)>,
524    pub failover_paths: Vec<Vec<String>>,
525    pub checksum: String,
526}
527
528impl ExecutionSchematic {
529    fn for_symbol(symbol: &Symbol, policy: &ConnectionPolicy) -> Self {
530        let base_node = format!("symbol:{}", symbol.id);
531        let mut nodes = vec![base_node.clone()];
532        let mut edges = Vec::new();
533        for zone in &policy.allowed_zones {
534            let zone_node = format!("zone:{}", zone);
535            if !nodes.contains(&zone_node) {
536                nodes.push(zone_node.clone());
537            }
538            edges.push((base_node.clone(), zone_node));
539        }
540        let failover_paths = if policy.allowed_zones.len() > 1 {
541            vec![policy.allowed_zones.iter().cloned().collect()]
542        } else {
543            vec![]
544        };
545        let checksum = format!(
546            "{}:{}:{}:{}",
547            symbol.schema_hash, symbol.version, policy.max_latency_ms, policy.min_trust_score
548        );
549        Self {
550            nodes,
551            edges,
552            failover_paths,
553            checksum,
554        }
555    }
556
557    fn incorporate_intent(&mut self, intent: &Intent) {
558        let intent_node = format!("intent:{}", intent.description);
559        if !self.nodes.contains(&intent_node) {
560            self.nodes.push(intent_node.clone());
561        }
562        for capability in &intent.required_capabilities {
563            let cap_node = format!("capability:{}", capability);
564            if !self.nodes.contains(&cap_node) {
565                self.nodes.push(cap_node.clone());
566            }
567            let edge = (intent_node.clone(), cap_node.clone());
568            if !self.edges.contains(&edge) {
569                self.edges.push(edge);
570            }
571        }
572        if self.failover_paths.is_empty() {
573            self.failover_paths
574                .push(vec![intent.target_kind.to_string()]);
575        }
576        self.checksum = format!("{}#{}", self.checksum, self.nodes.len());
577    }
578
579    fn merge(mut self, other: &ExecutionSchematic) -> ExecutionSchematic {
580        for node in &other.nodes {
581            if !self.nodes.contains(node) {
582                self.nodes.push(node.clone());
583            }
584        }
585        for edge in &other.edges {
586            if !self.edges.contains(edge) {
587                self.edges.push(edge.clone());
588            }
589        }
590        for path in &other.failover_paths {
591            if !self.failover_paths.contains(path) {
592                self.failover_paths.push(path.clone());
593            }
594        }
595        self.checksum = format!("{}+{}", self.checksum, other.checksum);
596        self
597    }
598}

642#[derive(Debug, Clone)]
643pub struct VerificationReport {
644    pub model_check_passed: bool,
645    pub smt_passed: bool,
646    pub failover_proved: bool,
647    pub issues: Vec<String>,
648}
649
650impl VerificationReport {
651    fn success() -> Self {
652        Self {
653            model_check_passed: true,
654            smt_passed: true,
655            failover_proved: true,
656            issues: Vec::new(),
657        }
658    }
659
660    fn evaluate(
661        intent: &Intent,
662        schematic: &ExecutionSchematic,
663        connectors: &[ConnectorSnapshot],
664        feeds: &HashMap<String, ReliabilityFeed>,
665    ) -> Self {
666        let mut issues = Vec::new();
667        let model_check_passed = connectors.iter().all(|snapshot| {
668            snapshot.state != ConnectionState::Faulted
669                && snapshot.health_score >= intent.constraints.min_trust_score
670        });
671        if !model_check_passed {
672            issues.push("connectors not healthy".to_string());
673        }
674
675        let smt_passed = connectors.iter().all(|snapshot| {
676            snapshot.latency_budget <= intent.constraints.max_latency_ms
677                && feeds
678                    .get(&snapshot.provider_id)
679                    .map(|feed| feed.probability_of_failure <= RELIABILITY_MICRO_THRESHOLD)
680                    .unwrap_or(true)
681        });
682        if !smt_passed {
683            issues.push("latency or reliability constraint violated".to_string());
684        }
685
686        let failover_proved =
687            schematic.failover_paths.iter().any(|path| path.len() >= 1) || connectors.len() > 1;
688        if !failover_proved {
689            issues.push("no failover path available".to_string());
690        }
691
692        Self {
693            model_check_passed,
694            smt_passed,
695            failover_proved,
696            issues,
697        }
698    }
699
700    fn is_success(&self) -> bool {
701        self.model_check_passed && self.smt_passed && self.failover_proved
702    }
703}

884
885    fn refresh(&mut self, now: SystemTime, feed: Option<&ReliabilityFeed>) -> ScanEvent {
886        let since_last = now.duration_since(self.last_seen).unwrap_or(Duration::ZERO);
887
888        if since_last > Duration::from_secs(5) {
889            self.health_score *= 0.95;
890            self.state = if self.health_score < 0.4 {
891                ConnectionState::Faulted
892            } else {
893                ConnectionState::Pending
894            };
895        } else {
896            self.health_score = (self.health_score + 1.0).min(1.0);
897            self.state = ConnectionState::Connected;
898        }
899
900        if !self.attestation_valid(now) {
901            self.health_score *= 0.9;
902            self.state = ConnectionState::Pending;
903            self.genome.record_signal("stale", -0.05);
904        } else {
905            self.health_score = (self.health_score + 0.08).min(1.0);
906            self.state = ConnectionState::Connected;
907            self.genome.record_signal("fresh", 0.06);
908        }
909
910        if let Some(feed) = feed {
911            if feed.is_degraded() {
912                self.health_score *= 0.9;
913                self.genome.record_signal("reliability", -0.08);
914            } else {
915                self.genome.record_signal("reliability", 0.04);
916            }
917        }
918
919        self.last_seen = now;
920
921        ScanEvent {
922            connector_id: self.symbol.id.clone(),
923            state: self.state,
924            health_score: self.health_score,
925        }
926    }
927
928    fn attestation_valid(&self, now: SystemTime) -> bool {
929        self.last_attestation
930            .as_ref()
931            .map(|record| {
932                let compliance_ok = if self.policy.required_compliance.is_empty() {
933                    true
934                } else {
935                    self.policy
936                        .required_compliance
937                        .is_subset(&record.proof.compliance_evidence)
938                };
939
940                let attestation_age = now
941                    .duration_since(record.validated_at)
942                    .unwrap_or(Duration::ZERO);
943                let attestation_fresh = attestation_age <= Duration::from_secs(120);
944
945                record.proof.is_fresh(now)
946                    && record.proof.confidence >= self.policy.min_attestation_score
947                    && compliance_ok
948                    && attestation_fresh
949            })
950            .unwrap_or(false)
951    }
952
953    fn adaptive_score(&self, signal: Option<&DemandSignal>, now: SystemTime) -> f32 {
954        let base = self.health_score;
955        let attestation_bonus = if self.attestation_valid(now) {
956            0.1
957        } else {
958            -0.2
959        };
960        let load_penalty = signal
961            .map(|s| {
962                let staleness = now.duration_since(s.last_updated).unwrap_or(Duration::ZERO);
963                let freshness = if staleness > Duration::from_secs(120) {
964                    0.0
965                } else {
966                    1.0 - (staleness.as_secs_f32() / 120.0)
967                };
968                ((1.0 - s.predicted_load.clamp(0.0, 1.0)) - 0.5) * freshness
969            })
970            .unwrap_or(0.0);
971        let feedback_bias: f32 = if self.recent_feedback.is_empty() {
972            0.0
973        } else {
974            let successes = self
975                .recent_feedback
976                .iter()
977                .filter(|result| **result)
978                .count() as f32;
979            (successes / self.recent_feedback.len() as f32) - 0.5
980        };
981
982        base + attestation_bonus + load_penalty + feedback_bias
983    }
984
985    fn record_feedback(&mut self, success: bool) {
986        if self.recent_feedback.len() == self.recent_feedback.capacity() {
987            self.recent_feedback.pop_front();
988        }
989        self.recent_feedback.push_back(success);
990        if success {
991            self.health_score = (self.health_score + 0.05).min(1.0);
992        } else {
993            self.health_score *= 0.92;
994        }
995    }
996
997    fn evolve(&mut self, evolution: &mut EvolutionState) -> Option<String> {
998        evolution.attempted += 1;
999        if let Some(mutation) = self.genome.propose_mutation(&self.symbol.capabilities) {
1000            evolution.accepted += 1;
1001            self.schematic.nodes.push(format!("genome:{}", mutation));
1002            return Some(mutation);
1003        }
1004        None
1005    }

2047    /// Calculate an optimized route for a given intent.
2048    pub fn route_intent(&self, intent: &Intent) -> Result<RoutePlan, GatewayError> {
2049        #[derive(Clone)]
2050        struct CandidateView {
2051            id: String,
2052            provider_id: String,
2053            max_latency: u32,
2054            snapshot: ConnectorSnapshot,
2055            schematic: ExecutionSchematic,
2056            score: f32,
2057        }
2058
2059        let reliability = {
2060            let feeds = self.reliability_read()?;
2061            feeds.clone()
2062        };
2063        let demand_signals = {
2064            let signals = self.demand_signals_read()?;
2065            signals.clone()
2066        };
2067        let now = SystemTime::now();
2068        let demand_signal = demand_signals.get(&intent.target_kind);
2069
2070        let connectors_guard = self.connectors_read()?;
2071        let mut candidates: Vec<CandidateView> = connectors_guard
2072            .values()
2073            .filter(|record| {
2074                record.symbol.kind == intent.target_kind
2075                    && record
2076                        .symbol
2077                        .matches_capabilities(&intent.required_capabilities)
2078                    && record.policy.allows(&intent.constraints)
2079                    && record.health_score >= intent.constraints.min_trust_score
2080                    && record.state != ConnectionState::Faulted
2081            })
2082            .map(|record| {
2083                let reliability_penalty = reliability
2084                    .get(&record.provider_id)
2085                    .map(|feed| feed.probability_of_failure)
2086                    .unwrap_or(0.0);
2087                let score = record.adaptive_score(demand_signal, now) - reliability_penalty;
2088                CandidateView {
2089                    id: record.symbol.id.clone(),
2090                    provider_id: record.provider_id.clone(),
2091                    max_latency: record.policy.max_latency_ms,
2092                    snapshot: record.snapshot(),
2093                    schematic: record.schematic.clone(),
2094                    score,
2095                }
2096            })
2097            .collect();
2098        drop(connectors_guard);
2099
2100        if candidates.is_empty() {
2101            return Err(GatewayError::NoRouteFound);
2102        }
2103
2104        candidates.sort_by(|a, b| {
2105            b.score
2106                .partial_cmp(&a.score)
2107                .unwrap_or(std::cmp::Ordering::Equal)
2108        });
2109
2110        let mut schematic = candidates[0].schematic.clone();
2111        schematic.incorporate_intent(intent);
2112        for view in candidates.iter().skip(1) {
2113            schematic = schematic.merge(&view.schematic);
2114        }
2115
2116        let selected: Vec<String> = candidates
2117            .iter()
2118            .take(3)
2119            .map(|view| view.id.clone())
2120            .collect();
2121        let provider_mix: Vec<String> = candidates
2122            .iter()
2123            .take(3)
2124            .map(|view| view.provider_id.clone())
2125            .collect();
2126
2127        let predicted_latency_ms = candidates
2128            .iter()
2129            .map(|view| view.max_latency)
2130            .min()
2131            .unwrap_or(intent.constraints.max_latency_ms);
2132
