19const DEFAULT_OUTPUT_DIR: &str = ".workspace/indexes";
20const AST_INDEX: &str = "ast_graph.json";
21const OWNERSHIP_INDEX: &str = "ownership_graph.json";
22const CONFIG_INDEX: &str = "config_graph.json";
23
24#[derive(Debug, Error)]
25pub enum IndexerError {
26    #[error("io error: {0}")]
27    Io(#[from] std::io::Error),
28    #[error("syntax error: {0}")]
29    Syntax(#[from] syn::Error),
30    #[error("config error: {0}")]
31    Config(#[from] toml::de::Error),
32    #[error("registry error: {0}")]
33    Registry(#[from] RegistryError),
34    #[error("serialization error: {0}")]
35    Serialization(#[from] serde_json::Error),
36}

38#[derive(Debug, Clone, Serialize, Deserialize)]
39pub struct IndexArtifacts {
40    pub generated_at: u128,
41    pub ast: AstGraph,
42    pub ownership: OwnershipGraph,
43    pub config: ConfigGraph,
44}
45
46pub struct IndexerService {
47    source: PathBuf,
48    output: PathBuf,
49}
50
51impl IndexerService {
52    pub fn new(source: impl AsRef<Path>) -> Self {
53        Self {
54            source: source.as_ref().to_path_buf(),
55            output: PathBuf::from(DEFAULT_OUTPUT_DIR),
56        }
57    }
58
59    pub fn for_workspace() -> Self {
60        Self::new(Path::new("."))
61    }
62
63    pub fn with_output_dir(mut self, output: impl AsRef<Path>) -> Self {
64        self.output = output.as_ref().to_path_buf();
65        self
66    }
67
68    pub fn refresh(&self) -> Result<IndexArtifacts, IndexerError> {
69        let ast = AstGraph::build(&self.source)?;
70        let ownership = OwnershipGraph::build()?;
71        let config = ConfigGraph::build(&self.source)?;
72
73        let artifacts = IndexArtifacts {
74            generated_at: crate::utils::current_timestamp_millis(),
75            ast,
76            ownership,
77            config,
78        };
79        self.persist(&artifacts)?;
80        Ok(artifacts)
81    }
82
83    pub fn persist(&self, artifacts: &IndexArtifacts) -> Result<(), IndexerError> {
84        fs::create_dir_all(&self.output)?;
85        write_json(self.output.join(AST_INDEX), &artifacts.ast)?;
86        write_json(self.output.join(OWNERSHIP_INDEX), &artifacts.ownership)?;
87        write_json(self.output.join(CONFIG_INDEX), &artifacts.config)?;
88        Ok(())
89    }
90}
91
92fn write_json(path: PathBuf, value: &impl Serialize) -> Result<(), IndexerError> {
93    let file = File::create(path)?;
94    let writer = BufWriter::new(file);
95    serde_json::to_writer_pretty(writer, value)?;
96    Ok(())
97}

104    #[test]
105    fn refresh_generates_indexes() {
106        let dir = tempdir().unwrap();
107        let service = IndexerService::new(Path::new("src")).with_output_dir(dir.path());
108        let artifacts = service.refresh().expect("indexing succeeds");
109        assert!(!artifacts.ast.nodes.is_empty());
110        assert!(dir.path().join(AST_INDEX).exists());
111        assert!(dir.path().join(OWNERSHIP_INDEX).exists());
112        assert!(dir.path().join(CONFIG_INDEX).exists());
