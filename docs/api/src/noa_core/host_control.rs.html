1//! Host control surface enabling environment takeover and resource arbitration.
2
3use std::collections::HashMap;
4use std::sync::{Mutex, OnceLock};
5
6use crate::config::manifest::{SCOPE_HOST_ENVIRONMENT_TAKEOVER, SCOPE_HOST_RESOURCE_ARBITRATE};
7use crate::time::current_timestamp_millis;
8use crate::token::{self, TokenError};
9
10/// Lease describing a token-bound environment takeover.
11#[derive(Debug, Clone, PartialEq, Eq)]
12pub struct EnvironmentLease {
13    pub environment: String,
14    pub token: String,
15    pub issued_to: String,
16    pub granted_at: u128,
17    pub expires_at: u128,
18}
19
20/// Request payload for resource arbitration decisions.
21#[derive(Debug, Clone, PartialEq)]
22pub struct ResourceArbitrationRequest {
23    pub environment: String,
24    pub desired_cpu_share: f32,
25    pub desired_memory_bytes: u64,
26}
27
28/// Arbitration decision containing the granted resource envelope.
29#[derive(Debug, Clone, PartialEq)]
30pub struct ResourceArbitrationDecision {
31    pub environment: String,
32    pub granted_cpu_share: f32,
33    pub granted_memory_bytes: u64,
34    pub isolation_enforced: bool,
35}
36
37/// Errors produced by the host control surface.
38#[derive(Debug, thiserror::Error)]
39pub enum HostControlError {
40    #[error(transparent)]
41    Token(#[from] TokenError),
42    #[error("environment {0} already leased by a different token")]
43    EnvironmentInUse(String),
44    #[error("environment {0} has no active lease")]
45    EnvironmentNotLeased(String),
46    #[error("environment {0} isolated from token")]
47    EnvironmentIsolationViolation(String),
48}
49
50#[derive(Debug, Default)]
51struct LeaseStore {
52    leases: HashMap<String, EnvironmentLease>,
53}
54
55/// Host control service responsible for coordinating isolation.
56pub struct HostControlService {
57    store: Mutex<LeaseStore>,
58}
59
60impl HostControlService {
61    fn new() -> Self {
62        Self {
63            store: Mutex::new(LeaseStore::default()),
64        }
65    }
66
67    /// Request an environment takeover using the provided token.
68    pub fn request_environment_takeover(
69        &self,
70        token: &str,
71        environment: impl Into<String>,
72    ) -> Result<EnvironmentLease, HostControlError> {
73        let environment = environment.into();
74        let validated = token::service().validate(token, SCOPE_HOST_ENVIRONMENT_TAKEOVER)?;
75        let mut store = self.store.lock().expect("lease store mutex poisoned");
76        if let Some(existing) = store.leases.get(&environment) {
77            if existing.token != token {
78                return Err(HostControlError::EnvironmentInUse(environment));
79            }
80            return Ok(existing.clone());
81        }
82
83        let lease = EnvironmentLease {
84            environment: environment.clone(),
85            token: validated.token.clone(),
86            issued_to: validated.issued_to.clone(),
87            granted_at: current_timestamp_millis(),
88            expires_at: validated.expires_at,
89        };
90        store.leases.insert(environment, lease.clone());
91        Ok(lease)
92    }
93
94    /// Release an active environment lease.
95    pub fn release_environment(
96        &self,
97        token: &str,
98        environment: impl Into<String>,
99    ) -> Result<(), HostControlError> {
100        let environment = environment.into();
101        let validated = token::service().validate(token, SCOPE_HOST_ENVIRONMENT_TAKEOVER)?;
102        let mut store = self.store.lock().expect("lease store mutex poisoned");
103        match store.leases.get(&environment) {
104            Some(existing) if existing.token == validated.token => {
105                store.leases.remove(&environment);
106                Ok(())
107            }
108            Some(_) => Err(HostControlError::EnvironmentIsolationViolation(environment)),
109            None => Err(HostControlError::EnvironmentNotLeased(environment)),
110        }
111    }
112
113    /// Arbitrate resource usage for an environment.
114    pub fn arbitrate_resources(
115        &self,
116        token: &str,
117        request: ResourceArbitrationRequest,
118    ) -> Result<ResourceArbitrationDecision, HostControlError> {
119        let validated = token::service().validate(token, SCOPE_HOST_RESOURCE_ARBITRATE)?;
120        let store = self.store.lock().expect("lease store mutex poisoned");
121        let lease = store
122            .leases
123            .get(&request.environment)
124            .ok_or_else(|| HostControlError::EnvironmentNotLeased(request.environment.clone()))?;
125
126        if lease.token != validated.token {
127            return Err(HostControlError::EnvironmentIsolationViolation(
128                request.environment,
129            ));
130        }
131
132        // Clamp CPU usage to 75% of requested share and memory to 80%.
133        let granted_cpu = request.desired_cpu_share.clamp(0.0, 1.0).min(0.75);
134        let granted_memory = ((request.desired_memory_bytes as f64) * 0.8) as u64;
135
136        Ok(ResourceArbitrationDecision {
137            environment: request.environment,
138            granted_cpu_share: granted_cpu,
139            granted_memory_bytes: granted_memory,
140            isolation_enforced: true,
141        })
142    }
143
144    /// Enumerate active leases.
145    pub fn active_leases(&self) -> Vec<EnvironmentLease> {
146        let store = self.store.lock().expect("lease store mutex poisoned");
147        store.leases.values().cloned().collect()
148    }
149
150    /// Reset the host control service (test helper).
151    pub fn reset(&self) {
152        let mut store = self.store.lock().expect("lease store mutex poisoned");
153        store.leases.clear();
154    }
155}
156
157fn global_host_control() -> &'static HostControlService {
158    static HOST_CONTROL: OnceLock<HostControlService> = OnceLock::new();
159    HOST_CONTROL.get_or_init(HostControlService::new)
160}
161
162/// Access the global host control service.
163pub fn service() -> &'static HostControlService {
164    global_host_control()
165}

167#[cfg(test)]
168mod tests {
169    use super::*;
170    use crate::config::manifest::KernelManifest;
171    use crate::token::{service as token_service, TokenIssuanceRequest};
172
173    fn setup() {
174        token_service().reset();
175        let manifest = KernelManifest::default();
176        token::configure_from_manifest(&manifest);
177        service().reset();
178    }
179
180    #[test]
181    fn lease_requires_valid_scope() {
182        setup();
183        let request = TokenIssuanceRequest::new(
184            "tester",
185            [
186                SCOPE_HOST_ENVIRONMENT_TAKEOVER,
187                SCOPE_HOST_RESOURCE_ARBITRATE,
188            ],
189        );
190        let token = token_service()
191            .issue_token(request)
192            .expect("token should issue");
193        let lease = service()
194            .request_environment_takeover(&token.token, "lab")
195            .expect("lease should be granted");
196        assert_eq!(lease.environment, "lab");
197    }
