1//! Inter-process communication (IPC) subsystem
2
3use std::collections::HashMap;
4use std::sync::{Mutex, OnceLock};
5
6pub type ChannelId = u64;
7
8#[derive(Debug, Clone)]
9pub struct Message {
10    pub from: u64,
11    pub to: u64,
12    pub data: Vec<u8>,
13}
14
15fn message_queues() -> &'static Mutex<HashMap<ChannelId, Vec<Message>>> {
16    static MESSAGE_QUEUES: OnceLock<Mutex<HashMap<ChannelId, Vec<Message>>>> = OnceLock::new();
17    MESSAGE_QUEUES.get_or_init(|| Mutex::new(HashMap::new()))
18}
19
20/// Initialize IPC subsystem
21pub fn init() -> Result<(), &'static str> {
22    println!("[IPC] Initializing inter-process communication...");
23    Ok(())
24}
25
26fn create_channel_inner(channel_id: ChannelId) -> Result<(), &'static str> {
27    let mut queues = message_queues().lock().unwrap();
28    queues.insert(channel_id, Vec::new());
29    Ok(())
30}
31
32fn send_message_inner(channel_id: ChannelId, message: Message) -> Result<(), &'static str> {
33    let mut queues = message_queues().lock().unwrap();
34    if let Some(queue) = queues.get_mut(&channel_id) {
35        queue.push(message);
36        Ok(())
37    } else {
38        Err("Channel not found")
39    }
40}
41
42fn receive_message_inner(channel_id: ChannelId) -> Option<Message> {
43    let mut queues = message_queues().lock().unwrap();
44    if let Some(queue) = queues.get_mut(&channel_id) {
45        if !queue.is_empty() {
46            Some(queue.remove(0))
47        } else {
48            None
49        }
50    } else {
51        None
52    }
53}
54
55/// Capability wrapper over IPC primitives.
56#[derive(Clone, Default)]
57pub struct IpcService;
58
59impl IpcService {
60    /// Register a new communication channel.
61    pub fn create_channel(&self, channel_id: ChannelId) -> Result<(), &'static str> {
62        create_channel_inner(channel_id)
63    }
64
65    /// Deliver a message to a channel.
66    pub fn send_message(
67        &self,
68        channel_id: ChannelId,
69        message: Message,
70    ) -> Result<(), &'static str> {
71        send_message_inner(channel_id, message)
72    }
73
74    /// Pull the next message from a channel queue.
75    pub fn receive_message(&self, channel_id: ChannelId) -> Option<Message> {
76        receive_message_inner(channel_id)
77    }
78}
79
80/// Create a new channel.
81pub fn create_channel(channel_id: ChannelId) -> Result<(), &'static str> {
82    IpcService::default().create_channel(channel_id)
83}
84
85/// Send a message.
86pub fn send_message(channel_id: ChannelId, message: Message) -> Result<(), &'static str> {
87    IpcService::default().send_message(channel_id, message)
88}
89
90/// Receive a message.
91pub fn receive_message(channel_id: ChannelId) -> Option<Message> {
92    IpcService::default().receive_message(channel_id)
93}
