11/// Internal counter used to produce deterministic token identifiers.
12fn issuance_counter() -> &'static AtomicU64 {
13    static COUNTER: OnceLock<AtomicU64> = OnceLock::new();
14    COUNTER.get_or_init(|| AtomicU64::new(1))
15}

17/// Token metadata describing an issued capability ticket.
18#[derive(Debug, Clone, PartialEq, Eq)]
19pub struct ScopeToken {
20    /// Unique token secret that must accompany capability requests.
21    pub token: String,
22    /// Actor receiving the token.
23    pub issued_to: String,
24    /// Scopes authorised by this token.
25    pub scopes: Vec<String>,
26    /// Timestamp (milliseconds) when the token was issued.
27    pub issued_at: u128,
28    /// Timestamp (milliseconds) when the token expires.
29    pub expires_at: u128,
30    /// Optional metadata captured at issuance time.
31    pub metadata: HashMap<String, String>,
32}
33
34impl ScopeToken {
35    /// Whether the token has expired relative to the provided timestamp.
36    pub fn is_expired(&self, timestamp_ms: u128) -> bool {
37        timestamp_ms >= self.expires_at
38    }
39
40    /// Whether the token grants the provided scope.
41    pub fn grants_scope(&self, scope: &str) -> bool {
42        self.scopes.iter().any(|candidate| candidate == scope)
43    }
44}

46/// Request payload describing a token to be issued.
47#[derive(Debug, Clone, PartialEq, Eq)]
48pub struct TokenIssuanceRequest {
49    actor: String,
50    scopes: Vec<String>,
51    ttl_override: Option<u64>,
52    metadata: HashMap<String, String>,
53}
54
55impl TokenIssuanceRequest {
56    /// Construct a new issuance request for the provided actor and scopes.
57    pub fn new(
58        actor: impl Into<String>,
59        scopes: impl IntoIterator<Item = impl Into<String>>,
60    ) -> Self {
61        Self {
62            actor: actor.into(),
63            scopes: scopes.into_iter().map(Into::into).collect(),
64            ttl_override: None,
65            metadata: HashMap::new(),
66        }
67    }
68
69    /// Override the maximum lifetime permitted by the policy.
70    pub fn with_ttl_seconds(mut self, ttl_seconds: u64) -> Self {
71        self.ttl_override = Some(ttl_seconds);
72        self
73    }
74
75    /// Attach metadata to the resulting token.
76    pub fn with_metadata(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
77        self.metadata.insert(key.into(), value.into());
78        self
79    }
80
81    /// Access the requested scopes.
82    pub fn scopes(&self) -> &[String] {
83        &self.scopes
84    }
85
86    /// Access the target actor.
87    pub fn actor(&self) -> &str {
88        &self.actor
89    }
90
91    /// Metadata map to embed in the token.
92    pub fn metadata(&self) -> &HashMap<String, String> {
93        &self.metadata
94    }
95}

97/// Policy describing a capability scope exposed via tokens.
98#[derive(Debug, Clone, PartialEq, Eq)]
99pub struct TokenPolicy {
100    pub scope: String,
101    pub description: Option<String>,
102    pub ttl_seconds: u64,
103    pub capabilities: Vec<String>,
104}
105
106impl From<TokenPolicyManifestEntry> for TokenPolicy {
107    fn from(entry: TokenPolicyManifestEntry) -> Self {
108        Self {
109            scope: entry.scope,
110            description: entry.description,
111            ttl_seconds: entry.ttl_seconds,
112            capabilities: entry.capabilities,
113        }
114    }
115}

117#[derive(Debug, Default)]
118struct TokenStore {
119    policies: HashMap<String, TokenPolicy>,
120    issued: HashMap<String, ScopeToken>,
121    revoked: HashSet<String>,
122}
123
124impl TokenStore {
125    fn configure(&mut self, policies: Vec<TokenPolicy>) {
126        self.policies = policies
127            .into_iter()
128            .map(|policy| (policy.scope.clone(), policy))
129            .collect();
130        self.issued.clear();
131        self.revoked.clear();
132    }
133}

135/// Errors emitted by the capability token service.
136#[derive(Debug, thiserror::Error)]
137pub enum TokenError {
138    #[error("token service has not been configured")]
139    NotConfigured,
140    #[error("scope {0} is not registered in the manifest")]
141    UnknownScope(String),
142    #[error("token {0} is not recognised")]
143    UnknownToken(String),
144    #[error("token {0} has been revoked")]
145    Revoked(String),
146    #[error("token expired at {0}")]
147    Expired(u128),
148    #[error("token missing required scope {0}")]
149    ScopeMissing(String),
150    #[error("issuance request does not contain any scopes")]
151    MissingScopes,
152    #[error("requested ttl {requested}s exceeds policy limit {policy}s for scope {scope}")]
153    TtlExceedsPolicy {
154        scope: String,
155        requested: u64,
156        policy: u64,
157    },
158}

160/// Global capability token service.
161pub struct CapabilityTokenService {
162    store: Mutex<TokenStore>,
163}
164
165impl CapabilityTokenService {
166    fn new() -> Self {
167        Self {
168            store: Mutex::new(TokenStore::default()),
169        }
170    }

172    /// Configure the service using the manifest token policies.
173    pub fn configure(&self, policies: Vec<TokenPolicyManifestEntry>) {
174        let policies: Vec<TokenPolicy> = policies.into_iter().map(TokenPolicy::from).collect();
175        let mut store = self.store.lock().expect("token store mutex poisoned");
176        store.configure(policies);
177    }
178
179    /// Reset the token store. Intended for tests.
180    pub fn reset(&self) {
181        let mut store = self.store.lock().expect("token store mutex poisoned");
182        *store = TokenStore::default();
183    }

185    /// Issue a new token based on the provided request.
186    pub fn issue_token(&self, request: TokenIssuanceRequest) -> Result<ScopeToken, TokenError> {
187        let mut store = self.store.lock().expect("token store mutex poisoned");
188        if store.policies.is_empty() {
189            return Err(TokenError::NotConfigured);
190        }
191        if request.scopes.is_empty() {
192            return Err(TokenError::MissingScopes);
193        }
194
195        let mut unique_scopes: Vec<String> = Vec::new();
196        let mut seen = HashSet::new();
197        for scope in request.scopes.iter() {
198            if seen.insert(scope.clone()) {
199                if !store.policies.contains_key(scope) {
200                    return Err(TokenError::UnknownScope(scope.clone()));
201                }
202                unique_scopes.push(scope.clone());
203            }
204        }
205
206        let now = current_timestamp_millis();
207        // Find the scope with the minimum TTL (ttl_ceiling)
208        let (ttl_ceiling, ttl_scope) = unique_scopes
209            .iter()
210            .filter_map(|scope| {
211                store
212                    .policies
213                    .get(scope)
214                    .map(|policy| (policy.ttl_seconds, scope))
215            })
216            .min_by_key(|(ttl, _scope)| *ttl)
217            .map(|(ttl, scope)| (ttl, scope.clone()))
218            .expect("at least one scope should be present");
219
220        if let Some(requested_ttl) = request.ttl_override {
221            if requested_ttl > ttl_ceiling {
222                return Err(TokenError::TtlExceedsPolicy {
223                    scope: ttl_scope,
224                    requested: requested_ttl,
225                    policy: ttl_ceiling,
226                });
227            }
228        }
229
230        let ttl_seconds = request.ttl_override.unwrap_or(ttl_ceiling);
231        let expires_at = now + (ttl_seconds as u128 * 1000);
232        let counter = issuance_counter().fetch_add(1, Ordering::SeqCst);
233        let token_secret = simple_hash(&format!("token::{}::{}::{}", request.actor, now, counter));
234
235        let token = ScopeToken {
236            token: token_secret.clone(),
237            issued_to: request.actor.clone(),
238            scopes: unique_scopes,
239            issued_at: now,
240            expires_at,
241            metadata: request.metadata.clone(),
242        };
243
244        store.issued.insert(token_secret, token.clone());
245        Ok(token)
246    }
247
248    /// Validate a token against the provided scope.
249    pub fn validate(&self, token: &str, scope: &str) -> Result<ScopeToken, TokenError> {
250        let now = current_timestamp_millis();
251        let store = self.store.lock().expect("token store mutex poisoned");
252        if store.policies.is_empty() {
253            return Err(TokenError::NotConfigured);
254        }
255        let issued = store
256            .issued
257            .get(token)
258            .cloned()
259            .ok_or_else(|| TokenError::UnknownToken(token.to_string()))?;
260
261        if store.revoked.contains(token) {
262            return Err(TokenError::Revoked(token.to_string()));
263        }
264        if issued.is_expired(now) {
265            return Err(TokenError::Expired(issued.expires_at));
266        }
267        if !issued.grants_scope(scope) {
268            return Err(TokenError::ScopeMissing(scope.to_string()));
269        }
270        Ok(issued)
271    }
272
273    /// Revoke a previously issued token.
274    pub fn revoke(&self, token: &str) -> Result<(), TokenError> {
275        let mut store = self.store.lock().expect("token store mutex poisoned");
276        if store.policies.is_empty() {
277            return Err(TokenError::NotConfigured);
278        }
279        if !store.issued.contains_key(token) {
280            return Err(TokenError::UnknownToken(token.to_string()));
281        }
282        store.revoked.insert(token.to_string());
283        Ok(())
284    }
285
286    /// List all configured scopes for diagnostics.
287    pub fn configured_scopes(&self) -> Vec<String> {
288        let store = self.store.lock().expect("token store mutex poisoned");
289        store.policies.keys().cloned().collect()
290    }
291}

293fn global_service() -> &'static CapabilityTokenService {
294    static SERVICE: OnceLock<CapabilityTokenService> = OnceLock::new();
295    SERVICE.get_or_init(CapabilityTokenService::new)
296}
297
298/// Access the global capability token service.
299pub fn service() -> &'static CapabilityTokenService {
300    global_service()
301}
302
303/// Configure the global token service using the manifest policies.
304pub fn configure_from_manifest(manifest: &crate::config::manifest::KernelManifest) {
305    service().configure(manifest.token_policies().to_vec());
