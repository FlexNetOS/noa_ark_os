31/// Core OS version
32pub const VERSION: &str = env!("CARGO_PKG_VERSION");
33
34/// Initialize the core OS using the kernel capability system.
35pub fn init() -> Result<capabilities::KernelHandle, kernel::KernelError> {
36    println!("NOA ARK OS Core v{}", VERSION);
37    println!("Initializing kernel-managed capabilities...");
38    let handle = kernel::init()?;
39    println!("Initializing core services...");
40
41    // Initialize subsystems (kernel already initialized above)
42    memory::init().map_err(|e| kernel::KernelError::Init(e.to_string()))?;
43    process::init().map_err(|e| kernel::KernelError::Init(e.to_string()))?;
44    ipc::init().map_err(|e| kernel::KernelError::Init(e.to_string()))?;
45    fs::init().map_err(|e| kernel::KernelError::Init(e.to_string()))?;
46    security::init().map_err(|e| kernel::KernelError::Init(e.to_string()))?;
47    gateway::init()
48        .map_err(|_| kernel::KernelError::Init("gateway initialization failed".to_string()))?;
49
50    indexer::IndexerService::for_workspace()
51        .refresh()
52        .map_err(|e| kernel::KernelError::Init(format!("workspace indexing failed: {}", e)))?;
53
54    println!("Core OS initialized successfully");
55    Ok(handle)
56}
