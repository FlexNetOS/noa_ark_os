22/// GPU family identifiers used to drive backend selection.
23#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
24pub enum GpuBackend {
25    Nvidia,
26    Amd,
27    Intel,
28    Apple,
29    Unknown,
30}
31
32impl GpuBackend {
33    #[cfg_attr(not(test), allow(dead_code))]
34    fn from_vendor_hint(vendor_id: Option<&str>, name: &str) -> Self {
35        let vendor = vendor_id.unwrap_or("").to_ascii_lowercase();
36        if vendor.contains("10de") || name.to_ascii_lowercase().contains("nvidia") {
37            GpuBackend::Nvidia
38        } else if vendor.contains("1002")
39            || vendor.contains("1022")
40            || name.to_ascii_lowercase().contains("amd")
41            || name.to_ascii_lowercase().contains("radeon")
42        {
43            GpuBackend::Amd
44        } else if vendor.contains("8086") || name.to_ascii_lowercase().contains("intel") {
45            GpuBackend::Intel
46        } else if vendor.contains("106b") || name.to_ascii_lowercase().contains("apple") {
47            GpuBackend::Apple
48        } else {
49            GpuBackend::Unknown
50        }
51    }
52
53    pub fn vendor_name(&self) -> &'static str {
54        match self {
55            GpuBackend::Nvidia => "NVIDIA",
56            GpuBackend::Amd => "AMD",
57            GpuBackend::Intel => "Intel",
58            GpuBackend::Apple => "Apple",
59            GpuBackend::Unknown => "Unknown",
60        }
61    }
62}

92/// Aggregated hardware view shared with higher system layers.
93#[derive(Debug, Clone, Serialize, Deserialize)]
94pub struct HardwareProfile {
95    pub cpu: CpuProfile,
96    pub memory: MemoryProfile,
97    pub gpus: Vec<GpuProfile>,
98    pub accelerators: Vec<AcceleratorProfile>,
99}
100
101impl HardwareProfile {
102    pub fn total_memory_gb(&self) -> f64 {
103        (self.memory.total_bytes as f64) / (1024.0 * 1024.0 * 1024.0)
104    }
105
106    pub fn available_memory_gb(&self) -> f64 {
107        (self.memory.available_bytes as f64) / (1024.0 * 1024.0 * 1024.0)
108    }
109
110    pub fn has_gpu(&self) -> bool {
111        !self.gpus.is_empty()
112    }
113}

115/// Detect hardware capabilities on the current host.
116pub fn detect_hardware_profile() -> HardwareProfile {
117    let mut system = System::new_all();
118    system.refresh_all();
119
120    let cpus = system.cpus();
121    let logical_cores = cpus.len();
122    let cpu = cpus
123        .first()
124        .map(|cpu| CpuProfile {
125            brand: cpu.brand().to_string(),
126            vendor: cpu.vendor_id().to_string(),
127            physical_cores: system.physical_core_count().unwrap_or(logical_cores),
128            logical_cores,
129            frequency_mhz: Some(cpu.frequency() as u64),
130        })
131        .unwrap_or(CpuProfile {
132            brand: "unknown".to_string(),
133            vendor: "unknown".to_string(),
134            physical_cores: system.physical_core_count().unwrap_or(1),
135            logical_cores: system.cpus().len().max(1),
136            frequency_mhz: None,
137        });
138
139    let memory = MemoryProfile {
140        total_bytes: system.total_memory() * 1024,
141        available_bytes: system.available_memory() * 1024,
142    };
143
144    let gpus = detect_gpus(&system);
145    let accelerators = detect_accelerators(&gpus);
146
147    HardwareProfile {
148        cpu,
149        memory,
150        gpus,
151        accelerators,
152    }
153}

155fn detect_gpus(_system: &System) -> Vec<GpuProfile> {
156    let mut gpus = Vec::new();
157
158    // Note: sysinfo graphics_cards() API not available in this version
159    // Falling back to nvidia-smi detection.
160
161    gpus.extend(query_nvidia_smi());
162
163    gpus
164}

166fn query_nvidia_smi() -> Vec<GpuProfile> {
167    let mut gpus = Vec::new();
168    let output = Command::new("nvidia-smi")
169        .args([
170            "--query-gpu=name,memory.total,driver_version",
171            "--format=csv,noheader,nounits",
172        ])
173        .output();
174
175    if let Ok(output) = output {
176        if output.status.success() {
177            let stdout = String::from_utf8_lossy(&output.stdout);
178            for line in stdout.lines() {
179                let parts: Vec<_> = line.split(',').map(|s| s.trim()).collect();
180                // Skip empty lines and lines with empty first field
181                // Also ensure we have at least the name field
182                if parts.is_empty() || parts[0].is_empty() {
183                    continue;
184                }
185
186                let name = parts[0].to_string();
187                let memory_total_bytes = parts
188                    .get(1)
189                    .and_then(|value| value.parse::<u64>().ok())
190                    .map(|mb| mb * 1024 * 1024);
191                let driver = parts.get(2).map(|s| s.to_string());
192
193                gpus.push(GpuProfile {
194                    name,
195                    backend: GpuBackend::Nvidia,
196                    memory_total_bytes,
197                    driver,
198                });
199            }
200        }
201    }
202
203    gpus
204}

206fn detect_accelerators(gpus: &[GpuProfile]) -> Vec<AcceleratorProfile> {
207    let mut accelerators = Vec::new();
208
209    for gpu in gpus {
210        accelerators.push(AcceleratorProfile {
211            kind: AcceleratorKind::Gpu,
212            vendor: Some(gpu.backend.vendor_name().to_string()),
213            model: Some(gpu.name.clone()),
214            details: gpu.driver.clone(),
215        });
216    }
217
218    if std::env::var("TPU_VISIBLE_DEVICES").is_ok() {
219        accelerators.push(AcceleratorProfile {
220            kind: AcceleratorKind::Tpu,
221            vendor: Some("Google".to_string()),
222            model: None,
223            details: Some("Detected TPU_VISIBLE_DEVICES".to_string()),
224        });
225    }
226
227    if std::env::var("NEURO_ACCELERATORS").is_ok() {
228        accelerators.push(AcceleratorProfile {
229            kind: AcceleratorKind::Npu,
230            vendor: None,
231            model: None,
232            details: Some("NEURO_ACCELERATORS environment variable".to_string()),
233        });
234    }
235
236    accelerators
237}
