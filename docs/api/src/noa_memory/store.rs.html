41#[derive(Debug, Serialize, Deserialize)]
42struct PersistedStore {
43    records: Vec<MemoryRecord>,
44}
45
46impl PersistedStore {
47    fn load(path: &Path) -> Result<Vec<MemoryRecord>, MemoryError> {
48        if !path.exists() {
49            return Ok(Vec::new());
50        }
51        let reader = BufReader::new(File::open(path)?);
52        let persisted: PersistedStore = serde_cbor::from_reader(reader)?;
53        Ok(persisted.records)
54    }
55
56    fn save(path: &Path, records: &[MemoryRecord]) -> Result<(), MemoryError> {
57        if let Some(parent) = path.parent() {
58            fs::create_dir_all(parent)?;
59        }
60        let writer = BufWriter::new(File::create(path)?);
61        let persisted = PersistedStore {
62            records: records.to_vec(),
63        };
64        serde_cbor::to_writer(writer, &persisted)?;
65        Ok(())
66    }
67}

69#[derive(Debug)]
70pub struct LongTermMemory {
71    path: PathBuf,
72    entries: RwLock<Vec<MemoryRecord>>,
73    next_id: AtomicU64,
74}
75
76impl LongTermMemory {
77    pub fn open(root: impl AsRef<Path>) -> Result<Self, MemoryError> {
78        let path = root.as_ref().join(LONG_TERM_FILE);
79        let records = PersistedStore::load(&path)?;
80        let next_id = records
81            .iter()
82            .map(|record| record.id)
83            .max()
84            .map(|max_id| max_id + 1)
85            .unwrap_or(1);
86        Ok(Self {
87            path,
88            entries: RwLock::new(records),
89            next_id: AtomicU64::new(next_id),
90        })
91    }
92
93    fn persist(&self) -> Result<(), MemoryError> {
94        let snapshot = self.entries.read().clone();
95        PersistedStore::save(&self.path, &snapshot)
96    }
97}

99impl MemoryStore for LongTermMemory {
100    fn append(
101        &self,
102        agent: &str,
103        role: MemoryRole,
104        content: &str,
105        metadata: HashMap<String, String>,
106        tags: Vec<String>,
107    ) -> Result<MemoryRecord, MemoryError> {
108        let id = self.next_id.fetch_add(1, Ordering::SeqCst);
109        let record = MemoryRecord::new(
110            id,
111            agent.to_string(),
112            role,
113            content.to_string(),
114            metadata,
115            tags,
116        );
117        {
118            let mut entries = self.entries.write();
119            entries.push(record.clone());
120        }
121        self.persist()?;
122        Ok(record)
123    }
124
125    fn incremental(
126        &self,
127        cursor: u64,
128        limit: usize,
129    ) -> Result<(Vec<MemoryRecord>, u64), MemoryError> {
130        let entries = self.entries.read();
131        let mut collected = Vec::new();
132        let mut last_seen = cursor;
133        for record in entries
134            .iter()
135            .filter(|record| record.id > cursor)
136            .take(limit)
137        {
138            collected.push(record.clone());
139            last_seen = record.id;
140        }
141        Ok((collected, last_seen))
142    }
143}

153impl SessionMemory {
154    pub fn open(
155        root: impl AsRef<Path>,
156        session_id: impl Into<String>,
157    ) -> Result<Self, MemoryError> {
158        let session_id = session_id.into();
159        let path = root
160            .as_ref()
161            .join(format!("{}_{}.cbor", SESSION_SUFFIX, session_id));
162        let records = PersistedStore::load(&path)?;
163        let next_id = records.last().map(|record| record.id + 1).unwrap_or(1);
164        Ok(Self {
165            session_id,
166            path,
167            entries: RwLock::new(records),
168            next_id: AtomicU64::new(next_id),
169        })
170    }
171
172    fn persist(&self) -> Result<(), MemoryError> {
173        let snapshot = self.entries.read().clone();
174        PersistedStore::save(&self.path, &snapshot)
175    }
176
177    pub fn session_id(&self) -> &str {
178        &self.session_id
179    }
180}

228pub fn build_retrieval(
229    long_term: &(impl MemoryStore + ?Sized),
230    session: &(impl MemoryStore + ?Sized),
231    cursor: MemoryCursor,
232    limit: usize,
233) -> Result<MemoryRetrieval, MemoryError> {
234    let start = std::time::Instant::now();
235    let (session_records, session_cursor) = session.incremental(cursor.session, limit)?;
236    let remaining = limit.saturating_sub(session_records.len());
237    let (long_term_records, long_cursor) = if remaining > 0 {
238        long_term.incremental(cursor.long_term, remaining)?
239    } else {
240        (Vec::new(), cursor.long_term)
241    };
242
243    let mut records = session_records;
244    records.extend(long_term_records);
245    let total_bytes = records.iter().map(MemoryRecord::content_len).sum();
246    let next_cursor = MemoryCursor {
247        long_term: long_cursor,
248        session: session_cursor,
249    };
250
251    Ok(MemoryRetrieval {
252        records,
253        next_cursor,
254        total_bytes,
255        took_ms: start.elapsed().as_millis(),
256    })
