9const DEFAULT_QUARANTINE_ROOT: &str = "archive/quarantine";
10const DEFAULT_ARCHIVE_ROOT: &str = "archive";
11const LEDGER_NAME: &str = "ledger.json";

13#[derive(Debug, Deserialize)]
14struct StatusMetadata {
15    #[serde(default)]
16    quarantined_at: Option<String>,
17}
18
19pub fn run() -> Result<()> {
20    let quarantine_root = env::var("QUARANTINE_ROOT").unwrap_or_else(|_| DEFAULT_QUARANTINE_ROOT.to_string());
21    let archive_root = env::var("ARCHIVE_ROOT").unwrap_or_else(|_| DEFAULT_ARCHIVE_ROOT.to_string());
22
23    let root_path = Path::new(&quarantine_root);
24    if !root_path.exists() {
25        return Ok(());
26    }
27
28    let mut relocated = 0usize;
29    for entry in fs::read_dir(root_path).with_context(|| format!("failed to read {quarantine_root}"))? {
30        let entry = entry?;
31        if !entry.file_type()?.is_dir() {
32            continue;
33        }
34        let bundle_path = entry.path();
35        if should_skip_bundle(&bundle_path)? {
36            continue;
37        }
38        if relocate_bundle(&bundle_path, &archive_root, &quarantine_root)? {
39            relocated += 1;
40        }
41    }
42
43    if relocated > 0 {
44        println!("Relocated {relocated} quarantine bundle(s).");
45    }
46
47    Ok(())
48}

50fn should_skip_bundle(bundle_path: &Path) -> Result<bool> {
51    let status = load_status(bundle_path)?;
52    let Some(status) = status else { return Ok(true); };
53    let Some(quarantined_at) = status.quarantined_at else {
54        return Ok(true);
55    };
56    let parsed = chrono::DateTime::parse_from_rfc3339(&quarantined_at)
57        .or_else(|_| chrono::DateTime::parse_from_str(&quarantined_at, "%Y-%m-%dT%H:%M:%SZ"))
58        .with_context(|| format!("unable to parse quarantined_at in {}", bundle_path.display()))?;
59    let age = Utc::now() - parsed.with_timezone(&Utc);
60    Ok(age < Duration::days(90))
61}
62
63fn load_status(bundle_path: &Path) -> Result<Option<StatusMetadata>> {
64    let status_path = bundle_path.join("status.yaml");
65    if !status_path.exists() {
66        return Ok(None);
67    }
68    let data = fs::read_to_string(&status_path)
69        .with_context(|| format!("failed to read {}", status_path.display()))?;
70    let parsed: StatusMetadata = serde_yaml::from_str(&data)
71        .with_context(|| format!("invalid YAML in {}", status_path.display()))?;
72    Ok(Some(parsed))
73}

75fn relocate_bundle(bundle_path: &Path, archive_root: &str, quarantine_root: &str) -> Result<bool> {
76    let bundle_name = bundle_path
77        .file_name()
78        .and_then(|s| s.to_str())
79        .ok_or_else(|| anyhow!("invalid bundle name for {}", bundle_path.display()))?;
80    let now = Utc::now();
81    let year = now.format("%Y");
82    let month = now.format("%m");
83    let archive_month_dir = Path::new(archive_root).join(year.to_string()).join(month.to_string());
84    let final_dir = archive_month_dir.join("quarantine");
85    fs::create_dir_all(&final_dir)
86        .with_context(|| format!("failed to create {}", final_dir.display()))?;
87
88    let destination = final_dir.join(bundle_name);
89    if destination.exists() {
90        return Ok(false);
91    }
92
93    fs::rename(bundle_path, &destination)
94        .with_context(|| format!("failed to move {} to {}", bundle_path.display(), destination.display()))?;
95
96    if let Err(err) = (|| -> Result<()> {
97        let tar_path = final_dir.join(format!("{bundle_name}.tar.zst"));
98        create_archive(&destination, &tar_path)?;
99        let sha = compute_sha(&tar_path)?;
100        update_ledger(
101            &archive_month_dir.join(LEDGER_NAME),
102            bundle_name,
103            &destination,
104            &tar_path,
105            &sha,
106            quarantine_root,
107        )?;
108        Ok(())
109    })() {
110        // Attempt to roll back the move on failure.
111        let _ = fs::rename(&destination, bundle_path);
112        return Err(err);
113    }
114    Ok(true)
115}

117fn create_archive(source_dir: &Path, archive_path: &Path) -> Result<()> {
118    let parent = source_dir
119        .parent()
120        .ok_or_else(|| anyhow!("missing parent for {}", source_dir.display()))?;
121    let dir_name = source_dir
122        .file_name()
123        .and_then(|s| s.to_str())
124        .ok_or_else(|| anyhow!("invalid directory name for {}", source_dir.display()))?;
125    if archive_path.exists() {
126        fs::remove_file(archive_path)
127            .with_context(|| format!("failed to remove existing {}", archive_path.display()))?;
128    }
129    let status = Command::new("tar")
130        .arg("--zstd")
131        .arg("-cf")
132        .arg(archive_path)
133        .arg("-C")
134        .arg(parent)
135        .arg(dir_name)
136        .status()
137        .with_context(|| format!("failed to create archive for {}", source_dir.display()))?;
138    if !status.success() {
139        return Err(anyhow!("tar command failed for {}", source_dir.display()));
140    }
141    Ok(())
142}

144fn compute_sha(path: &Path) -> Result<String> {
145    let data = fs::read(path).with_context(|| format!("failed to read {}", path.display()))?;
146    use sha2::{Digest, Sha256};
147    let mut hasher = Sha256::new();
148    hasher.update(&data);
149    Ok(format!("{:x}", hasher.finalize()))
150}

152fn update_ledger(
153    ledger_path: &Path,
154    bundle_name: &str,
155    destination: &Path,
156    tar_path: &Path,
157    sha: &str,
158    quarantine_root: &str,
159) -> Result<()> {
160    let mut entries: serde_json::Value = if ledger_path.exists() {
161        let raw = fs::read_to_string(ledger_path)
162            .with_context(|| format!("failed to read {}", ledger_path.display()))?;
163        serde_json::from_str(&raw).with_context(|| format!("invalid JSON in {}", ledger_path.display()))?
164    } else {
165        serde_json::json!([])
166    };
167    let array = entries
168        .as_array_mut()
169        .ok_or_else(|| anyhow!("ledger must be a JSON array: {}", ledger_path.display()))?;
170    let timestamp = Utc::now().format("%Y-%m-%dT%H:%M:%SZ").to_string();
171    array.push(serde_json::json!({
172        "type": "quarantine-rotation",
173        "bundle": destination.to_string_lossy(),
174        "archive": tar_path.to_string_lossy(),
175        "sha256": sha,
176        "timestamp": timestamp,
177        "source": format!("{}/{}", quarantine_root.trim_end_matches('/'), bundle_name),
178    }));
179    let mut file = fs::File::create(ledger_path)
180        .with_context(|| format!("failed to write {}", ledger_path.display()))?;
181    use std::io::Write;
182    file.write_all(serde_json::to_string_pretty(&entries)?.as_bytes())?;
183    file.write_all(b"\n")?;
184    Ok(())
185}
