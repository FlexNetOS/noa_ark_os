70impl UnifiedShell {
71    fn new(
72        platform: Platform,
73        modules: Vec<Arc<dyn ShellModule>>,
74        session: Option<UserSession>,
75    ) -> Result<Self, &'static str> {
76        let context = init(platform.clone())?;
77        let renderer = Renderer::new(context.clone());
78        let state = UIState::new(context.clone());
79        // Each shell instance holds its own store to avoid test interference when run in parallel.
80        let store = GlobalStore::new(GlobalState::default());
81
82        if let Some(session) = session {
83            store.update(|state| state.session = session);
84        } else {
85            store.update(|state| {
86                state.session = UserSession {
87                    user_id: "ops-admin".into(),
88                    display_name: "Ops Admin".into(),
89                    roles: vec![
90                        "admin".into(),
91                        "developer".into(),
92                        "operator".into(),
93                        "executive".into(),
94                    ],
95                    active_workspace: None,
96                    auth_token: None,
97                };
98            });
99        }
100
101        let workflow_catalog = WorkflowCatalog::default();
102        let event_log: Arc<Mutex<Vec<ShellEvent>>> = Arc::new(Mutex::new(vec![]));
103        let event_sink = {
104            let log = event_log.clone();
105            Arc::new(move |event: ShellEvent| {
106                log.lock().unwrap().push(event);
107            })
108        };
109
110        let services = ShellServices::new(store.clone(), event_sink.clone());
111        let chat_workspace = Arc::new(Mutex::new(ChatWorkspace::new(
112            store.clone(),
113            workflow_catalog.clone(),
114            event_sink.clone(),
115        )));
116
117        let shell = Self {
118            context,
119            renderer,
120            state,
121            store,
122            modules,
123            workflow_catalog,
124            chat_workspace,
125            analytics: Mutex::new(AnalyticsEngine::default()),
126            event_log,
127            services,
128        };
129
130        shell.bootstrap_modules(event_sink);
131        shell.seed_analytics();
132        shell.seed_knowledge_base();
133
134        Ok(shell)
135    }

137    fn bootstrap_modules(&self, event_sink: Arc<dyn Fn(ShellEvent) + Send + Sync>) {
138        let context = ModuleContext {
139            store: self.store.clone(),
140            workflows: self.workflow_catalog.clone(),
141            emit: event_sink.clone(),
142            services: self.services.clone(),
143        };
144
145        let mut commands = vec![];
146
147        for module in &self.modules {
148            module.hydrate(&context);
149            commands.extend(module.chat_commands());
150        }
151
152        if let Ok(mut chat) = self.chat_workspace.lock() {
153            for command in commands {
154                chat.register_command(command);
155            }
156
157            chat.register_command(crate::chat::ChatCommandDescriptor {
158                command: "continue work".into(),
159                description: "Continue the last development workflow".into(),
160                examples: vec!["continue work".into()],
161                action: crate::chat::ChatAction::TriggerWorkflow {
162                    workflow_id: "build".into(),
163                    payload: serde_json::json!({"resume": true}),
164                },
165            });
166        }
167
168        self.store.update(|state| {
169            state
170                .navigation
171                .primary_items
172                .sort_by(|a, b| a.label.cmp(&b.label));
173            if state.session.active_workspace.is_none() {
174                state.session.active_workspace = state.workspaces.keys().next().cloned();
175            }
176        });
177    }

296    pub fn emit(&self, event: ShellEvent) {
297        self.event_log.lock().unwrap().push(event.clone());
298        for module in &self.modules {
299            module.handle_event(
300                &event,
301                &ModuleContext {
302                    store: self.store.clone(),
303                    workflows: self.workflow_catalog.clone(),
304                    emit: Arc::new(|_| {}),
305                    services: self.services.clone(),
306                },
307            );
308        }
309    }
310
311    fn drain_events(&self) -> Vec<ShellEvent> {
312        let mut log = self.event_log.lock().unwrap();
313        let events = log.clone();
314        log.clear();
315        events
316    }

318    pub fn handle_chat_message(&self, message: &str) -> Option<String> {
319        let response = self
320            .chat_workspace
321            .lock()
322            .ok()
323            .and_then(|chat| chat.handle_message(message));
324
325        for event in self.drain_events() {
326            self.emit(event);
327        }
328
329        response
330    }

332    pub fn render(&self, adapter: &dyn PlatformAdapter) -> Result<(), String> {
333        let state_snapshot = self.store.read();
334        let session_roles = state_snapshot.session.roles.clone();
335        let mut nav_items: Vec<_> = state_snapshot
336            .navigation
337            .primary_items
338            .into_iter()
339            .filter(|item| {
340                item.allowed_roles.is_empty()
341                    || item
342                        .allowed_roles
343                        .iter()
344                        .any(|role| session_roles.iter().any(|r| r == role))
345            })
346            .collect();
347        nav_items.sort_by(|a, b| a.label.cmp(&b.label));
348
349        let mut active_route = state_snapshot
350            .navigation
351            .active_route
352            .clone()
353            .filter(|route| nav_items.iter().any(|item| &item.route == route));
354        if active_route.is_none() {
355            active_route = nav_items.first().map(|item| item.route.clone());
356        }
357
358        let mut workspaces: Vec<_> = state_snapshot
359            .workspaces
360            .values()
361            .cloned()
362            .filter(|workspace| {
363                workspace.allowed_roles.is_empty()
364                    || workspace
365                        .allowed_roles
366                        .iter()
367                        .any(|role| session_roles.iter().any(|r| r == role))
368            })
369            .collect();
370        workspaces.sort_by(|a, b| a.label.cmp(&b.label));
371
372        let mut active_workspace = state_snapshot
373            .session
374            .active_workspace
375            .clone()
376            .filter(|id| workspaces.iter().any(|workspace| &workspace.id == id));
377        if active_workspace.is_none() {
378            active_workspace = workspaces.first().map(|workspace| workspace.id.clone());
379        }
380
381        if let Some(active) = &active_workspace {
382            self.store.set_active_workspace(active.clone());
383        }
384
385        self.store.update(|state| {
386            state.navigation.active_route = active_route.clone();
387        });
388
389        let persona = active_workspace
390            .as_ref()
391            .and_then(|id| workspaces.iter().find(|workspace| &workspace.id == id))
392            .map(|workspace| workspace.persona)
393            .unwrap_or(WorkspacePersona::Developer);
394
395        let knowledge_articles = self.store.knowledge_for(persona);
396
397        let navigation = NavigationRail::new(nav_items, active_route.clone());
398        let workspace_switcher = WorkspaceSwitcher::new(workspaces, active_workspace.clone());
399        let knowledge = KnowledgeOverlay::new(persona, knowledge_articles);
400        let chrome = ShellChrome::new(navigation, workspace_switcher, knowledge);
401        adapter.mount(&self.renderer, &chrome, &self.state)?;
402        let frame = RenderFrame { chrome: &chrome };
403        self.renderer.render_frame(&frame)
404    }

406    pub fn recommended_adapter(&self) -> Box<dyn PlatformAdapter> {
407        match self.context.platform {
408            Platform::Server => Box::new(ServerAdapter),
409            Platform::Desktop => Box::new(TauriAdapter::default()),
410            Platform::Web => Box::new(ReactAdapter),
411            Platform::Mobile => Box::new(ReactNativeAdapter::default()),
412            Platform::ARGlasses | Platform::XRHeadset => Box::new(SpatialAdapter::default()),
413        }
414    }
